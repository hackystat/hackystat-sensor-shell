001    package org.hackystat.sensorshell;
002    
003    import static org.junit.Assert.assertEquals;
004    
005    import java.util.HashMap;
006    import java.util.Map;
007    
008    import javax.xml.datatype.XMLGregorianCalendar;
009    
010    import org.hackystat.sensorbase.client.SensorBaseClient;
011    import org.hackystat.sensorbase.resource.sensordata.Tstamp;
012    import org.hackystat.sensorbase.resource.sensordata.jaxb.SensorData;
013    import org.hackystat.sensorbase.server.Server;
014    import org.junit.BeforeClass;
015    import org.junit.Test;
016    
017    /**
018     * Provides simple unit tests for the SensorShell.
019     * @author Philip Johnson
020     */
021    public class TestSensorShell {
022      
023      /** The test user. */
024      private static String user = "TestShellUser@hackystat.org";
025      private static String host;
026      private static Server server;
027      
028      /**
029       * Starts the server going for these tests, and makes sure our test user is registered. 
030       * @throws Exception If problems occur setting up the server. 
031       */
032      @BeforeClass public static void setupServer() throws Exception {
033        TestSensorShell.server = Server.newInstance();
034        TestSensorShell.host = TestSensorShell.server.getHostName();
035        SensorBaseClient.registerUser(host, user);
036      }
037      
038      /**
039       * Tests that the SensorShell can send data to the Server and this data can be retrieved.
040       * @throws Exception If problems occur. 
041       */
042      @Test public void testSensorShell() throws Exception {
043        // First, start up a SensorShell.
044        SensorProperties properties = new SensorProperties(host, user, user);
045        // Create a SensorShell that is non-interactive, logging as "Test", and no offline storage. 
046        SensorShell shell = new SensorShell(properties, false, "Test", false);
047        
048        // Now construct a key-val map representing a SensorData instance. 
049        Map<String, String> keyValMap = new HashMap<String, String>();
050        XMLGregorianCalendar tstamp = Tstamp.makeTimestamp();
051        keyValMap.put("Timestamp", tstamp.toString());
052        String tool = "Eclipse";
053        keyValMap.put("Tool", tool);
054        keyValMap.put("SensorDataType", "DevEvent");
055        keyValMap.put("DevEvent-Type", "Compile");
056        
057        // Now add it to the SensorShell and send it to the server. 
058        shell.add(keyValMap);
059        shell.send();
060        shell.quit();
061        
062        // Now retrieve it from the server using a SensorBaseClient.
063        SensorBaseClient client = new SensorBaseClient(host, user, user);
064        // Now, get the data, dude. 
065        SensorData data = client.getSensorData(user, tstamp);
066        assertEquals("Checking data", tool, data.getTool());
067        // Now delete this data and the user for good measure.  
068        client.deleteSensorData(user, tstamp);
069        client.deleteUser(user);
070      }
071    
072    }
