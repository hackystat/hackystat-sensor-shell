001    package org.hackystat.sensorshell;
002    
003    import java.io.File;
004    import java.io.FileInputStream;
005    import java.io.FileOutputStream;
006    import java.text.SimpleDateFormat;
007    import java.util.ArrayList;
008    import java.util.Date;
009    import java.util.List;
010    import java.util.Locale;
011    import javax.xml.bind.JAXBContext;
012    import javax.xml.bind.Marshaller;
013    import javax.xml.bind.Unmarshaller;
014    import org.hackystat.sensorbase.resource.sensordata.jaxb.SensorData;
015    import org.hackystat.sensorbase.resource.sensordata.jaxb.SensorDatas;
016    import org.hackystat.utilities.home.HackystatUserHome;
017    
018    /**
019     * Provides a facility for persisting buffered SensorData instances locally when the SensorBase
020     * host is not available and then sending them during a subsequent invocation of SensorShell.
021     * 
022     * @author Philip Johnson
023     */
024    public class OfflineManager {
025      
026      /** The directory where offline data is stored. */
027      private File offlineDir;
028      
029      /** The SensorShell associated with this OfflineManager. */
030      private SingleSensorShell shell;
031      /** The jaxb context. */
032      private JAXBContext jaxbContext;
033      
034      /**
035       * Constructor for OfflineManager which initializes the location for offline data.
036       * @param shell The SensorShell associated with this OfflineManager.
037       */
038      public OfflineManager(SingleSensorShell shell) {
039        this.offlineDir = new File(HackystatUserHome.getHome(),  "/.hackystat/sensorshell/offline/");
040        this.offlineDir.mkdirs();
041        this.shell = shell;
042        try {
043          this.jaxbContext = 
044            JAXBContext.newInstance(
045                org.hackystat.sensorbase.resource.sensordata.jaxb.ObjectFactory.class);
046        }
047        catch (Exception e) {
048          throw new RuntimeException("Could not create JAXB context.", e);
049        }
050      }
051      
052      /**
053       * Stores a SensorDatas instance to a serialized file in the offline directory.
054       * @param sensorDatas The SensorDatas instance to be stored. 
055       */
056      public void store(SensorDatas sensorDatas) {
057        SimpleDateFormat fileTimestampFormat = 
058          new SimpleDateFormat("yyyy.MM.dd.HH.mm.ss.SSS", Locale.US);
059        String fileStampString = fileTimestampFormat.format(new Date());
060        File outFile = new File(this.offlineDir, fileStampString + ".xml");
061        try {
062          Marshaller marshaller = jaxbContext.createMarshaller();
063          marshaller.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, true);
064          marshaller.marshal(sensorDatas, new FileOutputStream(outFile));
065          }
066        catch (Exception e) {
067          shell.println("Error writing the offline file " + outFile.getName() + " " + e);
068        }
069        shell.println("Stored " + sensorDatas.getSensorData().size() + " sensor data instances in: " 
070            + outFile.getAbsolutePath());
071      }
072      
073      /**
074       * Recovers any previously stored SensorDatas instances from their serialized files.
075       * Each SensorDatas instance is deserialized, then each SensorData instance inside is
076       * individually sent to the SensorShell.  This gives the SensorShell an opportunity to 
077       * send batches off at whatever interval it chooses.
078       * All serialized files are deleted after being processed, regardless of whether or not
079       * processing is successful or not. 
080       * @throws SensorShellException If problems occur sending the recovered data.
081       */
082      public void recover() throws SensorShellException {
083        File[] xmlFiles = this.offlineDir.listFiles(new ExtensionFileFilter(".xml"));
084        FileInputStream fileStream = null;
085        List<SensorData> offlineData = new ArrayList<SensorData>(500);
086        for (int i = 0; i < xmlFiles.length; i++) {
087          try {
088            // Reconstruct the SensorDatas instances from the serialized files. 
089            this.shell.println("Recovering offline data from: " + xmlFiles[i].getName());
090            fileStream = new FileInputStream(xmlFiles[i]);
091            Unmarshaller unmarshaller = this.jaxbContext.createUnmarshaller();
092            SensorDatas sensorDatas = (SensorDatas)unmarshaller.unmarshal(fileStream);
093            // Now that we have the sensor data in memory, close and delete the file. 
094            fileStream.close();
095            xmlFiles[i].delete();
096            // Next, add this data to our local arraylist.
097            for (SensorData data : sensorDatas.getSensorData()) {
098              offlineData.add(data);
099            }
100            // Now, if we've collected enough local entries, send them out.
101            if (offlineData.size() > 500) {
102              this.shell.println("Sending recovered offline data (" + offlineData.size() + " entries)");
103              for (SensorData data : offlineData) {
104                shell.add(data);
105              }
106              offlineData.clear();
107            }
108          }
109          catch (Exception e) {
110            this.shell.println("Error recovering offline data from: " + xmlFiles[i] + " " + e);
111            try {
112              fileStream.close();
113            }
114            catch (Exception f) { 
115              this.shell.println("Failed to close: " + fileStream.toString() + " " + e);
116            }
117          }
118        }
119        // Clear out any remaining data from our local buffer. 
120        for (SensorData data : offlineData) {
121          shell.add(data);
122        }
123        this.shell.println("Explicit send of recovered data.");
124        shell.send();
125      }
126    }
