191      /**
192       * Creates a SensorShellProperties instance using the specified properties file. 
193       * All unspecified properties are set to their built-in default values. 
194       * @param sensorFile The sensorshell properties file to read.
195       * @throws SensorShellException If the SensorProperties instance cannot be 
196       * instantiated due to a missing host, user, and/or password properties.
197       */
198      public SensorShellProperties(File sensorFile) throws SensorShellException {
199        this.sensorShellPropertiesFile = sensorFile;
200        setDefaultSensorShellProperties(true);
201        FileInputStream fileStream = null;
202        try {
203          fileStream = new FileInputStream(this.sensorShellPropertiesFile);
204          this.sensorProps.load(fileStream);
205          validateProperties();
206        } 
207        catch (Exception e) {
208          String errMsg = "SensorShellProperties error: sensorFile is: " + sensorFile;
209          this.logger.warning(errMsg);
210          throw new SensorShellException(errMsg, e);
211        }
212        finally {
213          try {
214            if (fileStream != null) {
215              fileStream.close();
216            }
217          }
218          catch (Exception e) { //NOPMD
219            // Don't say anything. 
220          }
221        }
222      }

225       * Constructs a "basic" instance with the supplied three required properties.
226       * All other properties are assigned values from sensorshell.properties, or the 
227       * built-in defaults if not specified there.
228       * <p>
229       * Use SensorShellProperties.getTestInstance to create an instance for testing purposes, since
230       * it will override certain properties that may be present in the sensorshell.properties file.
231       * @param host The hackystat host.
232       * @param email The user's email.

245      }

333      /**
334       * Creates a SensorShell properties instance, initializing it using the passed properties as 
335       * well as any settings found in the sensorshell.properties file. 
336       * If overrideFile is true, then the passed properties and settings will override matching 
337       * properties and settings found in the sensorshell.properties file.  If false, then the 
338       * property settings found in the sensorshell.properties will override the passed properties.
339       * <p>
340       * This constructor enables the user to provide default settings in sensorshell.properties for 
341       * things like MultiSensorShell, but enable a tool to provide command line args that could
342       * override these defaults. The tool could do this by processing its command line args, then
343       * creating a Properties instance containing the overridding values, then passing that in to 
344       * this constructor.
345       * <p>
346       * If overrideFile is false, then these properties will only take effect if the user has not
347       * specified them in their sensorshell.properties file. 
348       * <p>
349       * Standard properties not specified by either the sensorshell.properties file or the passed 
350       * properties instance will be given default values.
351       *  
352       * @param properties The properties. 
353       * @param overrideFile If true, then the passed properties will override any matching 
354       * sensorshell.properties properties.
355       * @throws SensorShellException if problems occur.
356       */
357      public SensorShellProperties(Properties properties, boolean overrideFile) 
358      throws SensorShellException {
359        if (overrideFile) {
360          this.setPropertiesFromFile(sensorShellPropertiesFile);
361          this.sensorProps.putAll(properties);
362          this.setDefaultSensorShellProperties(false);
363        }
364        else {
365          this.sensorProps.putAll(properties);
366          this.setPropertiesFromFile(sensorShellPropertiesFile);
367          this.setDefaultSensorShellProperties(false);
368        }
369        validateProperties();
370      }

430      /**
431       * Checks that the standard sensor properties have valid values.
432       * <p>
433       * The approach is that the instance variables (timeout, multishellEnabled, etc.) always start off
434       * holding the default values, while the properties start off with potentially new and 
435       * potentially invalid values.  We attempt to set the instance variables to the new values
436       * in the properties.  If this fails, we use the instance variables to reset the properties
437       * to the defaults.
438       * <p>
439       * The approach for the required variables is that if they do not have values, we throw
440       * an error.   
441       * @throws SensorShellException if problems occur.
442       */
443      private final void validateProperties() throws SensorShellException {
444        String newValue = null;
445        String origValue = null;
446        String errMsg = "SensorProperties instantiation error: "; 
447        // TIMEOUT
448        try {
449          origValue = String.valueOf(this.timeout);
450          newValue = this.getProperty(SENSORSHELL_TIMEOUT_KEY);
451          this.timeout = Integer.parseInt(newValue);
452          if (this.timeout < 1) {
453            this.logger.warning(errMsg + SENSORSHELL_TIMEOUT_KEY + " " + newValue); 
454            this.sensorProps.setProperty(SENSORSHELL_TIMEOUT_KEY, origValue);
455            this.timeout = Integer.parseInt(origValue);
456          }
457        }
458        catch (Exception e) {
459          this.logger.warning(errMsg + SENSORSHELL_TIMEOUT_KEY + " " + newValue); 
460          this.sensorProps.setProperty(SENSORSHELL_TIMEOUT_KEY, origValue);
461        }
462        // MULTISHELL_ENABLED
463        try {
464          origValue = String.valueOf(multiShellEnabled);
465          newValue = this.getProperty(SENSORSHELL_MULTISHELL_ENABLED_KEY);
466          this.multiShellEnabled = Boolean.parseBoolean(newValue);
467        }
468        catch (Exception e) {
469          this.logger.warning(errMsg + SENSORSHELL_MULTISHELL_ENABLED_KEY + " " + newValue); 
470          this.sensorProps.setProperty(SENSORSHELL_MULTISHELL_ENABLED_KEY, origValue); 
471        }
472        // MULTISHELL_NUMSHELLS
473        try {
474          origValue = String.valueOf(multiShellNumShells);
475          newValue = this.getProperty(SENSORSHELL_MULTISHELL_NUMSHELLS_KEY);
476          this.multiShellNumShells = Integer.parseInt(newValue);
477          if (this.multiShellNumShells < 1) {
478            this.logger.warning(errMsg + SENSORSHELL_MULTISHELL_NUMSHELLS_KEY + " " + newValue); 
479            this.sensorProps.setProperty(SENSORSHELL_MULTISHELL_NUMSHELLS_KEY, origValue);
480            this.multiShellNumShells = Integer.parseInt(origValue);
481          }
482        }
483        catch (Exception e) {
484          this.logger.warning(errMsg + SENSORSHELL_MULTISHELL_NUMSHELLS_KEY + " " + newValue); 
485          this.sensorProps.setProperty(SENSORSHELL_MULTISHELL_NUMSHELLS_KEY, origValue); 
486        }
487        // MULTISHELL_BATCHSIZE
488        try {
489          origValue = String.valueOf(multiShellBatchSize);
490          newValue = this.getProperty(SENSORSHELL_MULTISHELL_BATCHSIZE_KEY);
491          this.multiShellBatchSize = Integer.parseInt(newValue);
492          if (this.multiShellBatchSize < 1) {
493            this.logger.warning(errMsg + SENSORSHELL_MULTISHELL_BATCHSIZE_KEY + " " + newValue); 
494            this.sensorProps.setProperty(SENSORSHELL_MULTISHELL_BATCHSIZE_KEY, origValue);
495            this.multiShellBatchSize = Integer.parseInt(origValue);
496          }
497        }
498        catch (Exception e) {
499          this.logger.warning(errMsg + SENSORSHELL_MULTISHELL_BATCHSIZE_KEY + " " + newValue); 
500          this.sensorProps.setProperty(SENSORSHELL_MULTISHELL_BATCHSIZE_KEY, origValue);
501        }
502        // MULTISHELL_AUTOSEND_TIMEINTERVAL
503        try {
504          origValue = String.valueOf(multiShellAutoSendTimeInterval);
505          newValue = this.getProperty(SENSORSHELL_MULTISHELL_AUTOSEND_TIMEINTERVAL_KEY);
506          this.multiShellAutoSendTimeInterval = Double.parseDouble(newValue);
507          if (this.multiShellAutoSendTimeInterval < 0.0) {
508            this.logger.warning(errMsg + SENSORSHELL_MULTISHELL_AUTOSEND_TIMEINTERVAL_KEY + newValue); 
509            this.sensorProps.setProperty(SENSORSHELL_MULTISHELL_AUTOSEND_TIMEINTERVAL_KEY, origValue);
510            this.multiShellAutoSendTimeInterval = Double.parseDouble(origValue);
511          }
512        }
513        catch (Exception e) {
514          this.logger.warning(errMsg + SENSORSHELL_MULTISHELL_AUTOSEND_TIMEINTERVAL_KEY + newValue); 
515          this.sensorProps.setProperty(SENSORSHELL_MULTISHELL_AUTOSEND_TIMEINTERVAL_KEY, origValue);
516        }
517        // OFFLINE_CACHE_ENABLED
518        try {
519          origValue = String.valueOf(offlineCacheEnabled);
520          newValue = this.getProperty(SENSORSHELL_OFFLINE_CACHE_ENABLED_KEY);
521          this.offlineCacheEnabled = Boolean.parseBoolean(newValue);
522        }
523        catch (Exception e) {
524          this.logger.warning(errMsg + SENSORSHELL_OFFLINE_CACHE_ENABLED_KEY + " " + newValue); 
525          this.sensorProps.setProperty(SENSORSHELL_OFFLINE_CACHE_ENABLED_KEY, origValue);
526        }
527        // OFFLINE_RECOVERY_ENABLED
528        try {
529          origValue = String.valueOf(offlineRecoveryEnabled);
530          newValue = this.getProperty(SENSORSHELL_OFFLINE_RECOVERY_ENABLED_KEY);
531          this.offlineRecoveryEnabled = Boolean.parseBoolean(newValue);
532        }
533        catch (Exception e) {
534          this.logger.warning(errMsg + SENSORSHELL_OFFLINE_RECOVERY_ENABLED_KEY + " " + newValue); 
535          this.sensorProps.setProperty(SENSORSHELL_OFFLINE_RECOVERY_ENABLED_KEY, origValue);
536        }
537        // STATECHANGE_INTERVAL
538        try {
539          origValue = String.valueOf(statechangeInterval);
540          newValue = this.getProperty(SENSORSHELL_STATECHANGE_INTERVAL_KEY);
541          this.statechangeInterval = Integer.parseInt(newValue);
542          if (this.statechangeInterval < 1) {
543            this.logger.warning(errMsg + SENSORSHELL_STATECHANGE_INTERVAL_KEY + " " + newValue); 
544            this.sensorProps.setProperty(SENSORSHELL_STATECHANGE_INTERVAL_KEY, origValue);
545            this.statechangeInterval = Integer.parseInt(origValue);
546          }
547        }
548        catch (Exception e) {
549          this.logger.warning(errMsg + SENSORSHELL_STATECHANGE_INTERVAL_KEY + " " + newValue); 
550          this.sensorProps.setProperty(SENSORSHELL_STATECHANGE_INTERVAL_KEY, origValue);
551        }
552        // AUTOSEND_TIMEINTERVAL (Single Shell)
553        try {
554          origValue = String.valueOf(autosendTimeInterval);
555          newValue = this.getProperty(SENSORSHELL_AUTOSEND_TIMEINTERVAL_KEY);
556          this.autosendTimeInterval = Double.parseDouble(newValue);
557          if (this.autosendTimeInterval < 0.0) {
558            this.logger.warning(errMsg + SENSORSHELL_AUTOSEND_TIMEINTERVAL_KEY + " " + newValue); 
559            this.sensorProps.setProperty(SENSORSHELL_AUTOSEND_TIMEINTERVAL_KEY, origValue);
560            this.autosendTimeInterval = Double.parseDouble(origValue);
561          }
562        }
563        catch (Exception e) {
564          this.logger.warning(errMsg + SENSORSHELL_AUTOSEND_TIMEINTERVAL_KEY + " " + newValue); 
565          this.sensorProps.setProperty(SENSORSHELL_AUTOSEND_TIMEINTERVAL_KEY, origValue);
566        }
567        // AUTOSEND_MAXBUFFER
568        try {
569          origValue = String.valueOf(autosendMaxBuffer);
570          newValue = this.getProperty(SENSORSHELL_AUTOSEND_MAXBUFFER_KEY);
571          this.autosendMaxBuffer = Integer.parseInt(newValue);
572          if (this.autosendMaxBuffer < 1) {
573            this.logger.warning(errMsg + SENSORSHELL_AUTOSEND_MAXBUFFER_KEY + " " + newValue); 
574            this.sensorProps.setProperty(SENSORSHELL_AUTOSEND_MAXBUFFER_KEY, origValue);
575            this.autosendMaxBuffer = Integer.parseInt(origValue);
576          }
577        }
578        catch (Exception e) {
579          this.logger.warning(errMsg + SENSORSHELL_AUTOSEND_MAXBUFFER_KEY + " " + newValue); 
580          this.sensorProps.setProperty(SENSORSHELL_AUTOSEND_MAXBUFFER_KEY, origValue);
581        }
582        // LOGGING_LEVEL
583        try {
584          origValue = String.valueOf(loggingLevel);
585          newValue = this.getProperty(SENSORSHELL_LOGGING_LEVEL_KEY);
586          this.loggingLevel = Level.parse(newValue);
587        }
588        catch (Exception e) {
589          this.logger.warning(errMsg + SENSORSHELL_LOGGING_LEVEL_KEY + " " + newValue); 
590          this.sensorProps.setProperty(SENSORSHELL_LOGGING_LEVEL_KEY, origValue);
591        }
592        
593        String errInfo = " You might wish to check the settings in " + 
594        sensorShellPropertiesFile.getAbsolutePath();
595        
596        
597        // SENSORBASE_HOST
598        this.sensorBaseHost = this.getProperty(SENSORSHELL_SENSORBASE_HOST_KEY);
599        if (this.sensorBaseHost == null) {
600          throw new SensorShellException("Missing sensorshell.sensorbase.host." + errInfo);
601        }
602        if (!this.sensorBaseHost.endsWith("/")) {
603          this.sensorBaseHost = this.sensorBaseHost + "/";
604        }
605        // SENSORBASE_USER
606        this.user = this.getProperty(SENSORSHELL_SENSORBASE_USER_KEY);
607        if (this.user == null) {
608          throw new SensorShellException("Missing sensorshell.sensorbase.user." + errInfo);
609        }
610        // SENSORBASE_PASSWORD
611        this.password = this.getProperty(SENSORSHELL_SENSORBASE_PASSWORD_KEY);
612        if (this.password == null) {
613          throw new SensorShellException("Missing sensorshell.sensorbase.password." + errInfo);
614        }
615      }

617      /**
618       * Ensures that the specified property has a value in this sensorshell properties.
619       * If overridePreexisting is true, then property will be set to value regardless of whether
620       * it had a previously existing value. 
621       * If overridePreexisting is false, then property will be set to value only if there is no
622       * currently defined property.    
623       * @param property The property to set, if not yet defined.
624       * @param value The value to set the property to, if not yet defined.
625       * @param overridePreexisting True if property will be set to value even if it currently exists. 
626       */
627      private void setDefaultProperty(String property, String value, boolean overridePreexisting) {
628        if (overridePreexisting) {
629          this.sensorProps.setProperty(property, value);
630        }
631        else if (!this.sensorProps.containsKey(property)) {
632          this.sensorProps.setProperty(property, value);
633        }
634      }

638       * clients who want to look for "non-standard" properties supplied in sensorshell.properties.
639       * For standard properties, clients should use the accessor methods, since these will perform
640       * type conversion.
641       * @param key The parameter key.

767      /**
768       * NOTE: We are not currently adding these values to the System properties object. The reason
769       * is that there can be multiple SensorShell (and the SensorShellProperties) instances active
770       * within a single JVM, either due to MultiShell or UserMaps.  These can have potentially
771       * different properties settings, leading to unpredictable values in the System Properties
772       * instance. Thus, it is safer to not rely on the System Properties instance to store any
773       * of these values. 
774       * 
775       * Updates the System properties object with the contents of the passed Hackystat
776       * properties instance.  This method is declared static so that it can be invoked
777       * from both SensorProperties and ServerProperties.  The hackystat properties added from
778       * SensorProperties were found in sensor.properties. The hackystat properties added from
779       * ServerProperties were found in hackystat.server.properties.  
780       * This method is package-private for access by SensorProperties.
781       * @param hackyProperties A Properties instance containing hackystat properties.
782       */
783      @SuppressWarnings("unused")
784      private void addToSystemProperties(Properties hackyProperties) {
785        Properties systemProperties = System.getProperties();
786        systemProperties.putAll(hackyProperties);
787        System.setProperties(systemProperties);
788      }

790      /**
791       * Returns the set of SensorProperties as a multi-line string.
792       * Does not print out the value associated with SENSORSHELL_PASSWORD_KEY. 
793       * @return The Sensor property keys and values. 
794       */
795      @Override
796      public String toString() {
797        StringBuffer buff = new StringBuffer(100);
798        // It turns out to be much, much more usable to get these properties alphabetized.
799        TreeMap<String, String> map = new TreeMap<String, String>();
800        for (Object key : this.sensorProps.keySet()) {
801          map.put(key.toString(), this.sensorProps.get(key).toString());
802        }
803        // Now print them out in alphabetical order.
804        buff.append("SensorProperties");
805        for (Entry<String, String> entry : map.entrySet()) {
806          buff.append("\n  ");
807          buff.append(entry.getKey());
808          buff.append(" : "); 
809          buff.append((entry.getKey().equals(SENSORSHELL_SENSORBASE_PASSWORD_KEY)) ? 
810              "<password hidden>" : entry.getValue());
811        }
812        buff.append("\n  sensorshell.properties file location: ");
813        buff.append(this.sensorShellPropertiesFile.getAbsolutePath());
814        return buff.toString();
815      }
