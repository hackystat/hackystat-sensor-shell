001    package org.hackystat.sensorshell;
002    
003    import java.io.File;
004    import java.io.FileInputStream;
005    import java.io.FileNotFoundException;
006    import java.io.IOException;
007    import java.util.Properties;
008    
009    /**
010     * Provides access to the hackystat properties file for each sensor, and provides reasonable default
011     * values when properties file is missing, cannot be read, or lacks property values. <p>
012     *
013     * The sensor properties file is stored in userhome/.hackystat/v8.sensor.properties. It is a
014     * user-maintained file of key=value pairs. <p>
015     * 
016     * All of the properties in sensor.properties are added to the System property instance after being
017     * read from the file.  This enables other tools that interoperate with Hackystat to obtain 
018     * settings from the System instance and/or environments to configure System property settings
019     * (such as for SSL) simply by adding entries to the properties file. 
020     *
021     * @author    Philip M. Johnson
022     */
023    public class SensorProperties {
024    
025      /** Hackystat host. */
026      private static final String HOST_KEY = "HACKYSTAT_SENSORBASE_HOST";
027      /** User's email. */
028      private static final String EMAIL_KEY = "HACKYSTAT_EMAIL";
029      /** User's password. */
030      private static final String PASSWORD_KEY = "HACKYSTAT_PASSWORD";
031      /** Auto send interval.  */
032      private static final String AUTOSEND_KEY = "HACKYSTAT_AUTOSEND_INTERVAL";
033      /** State change interval. */
034      private static final String STATECHANGE_KEY = "HACKYSTAT_STATE_CHANGE_INTERVAL";
035    
036      /** The standard location of the sensor properties file. */
037      private File sensorFile;
038    
039      /** Whether the properties file exists and was successfully read. */
040      private boolean fileAvailable = false;
041      /** The internal properties object. */
042      private Properties sensorProps = new Properties();
043    
044    
045    
046      /**
047       * Initializes based on the user's v8.sensor.properties file.
048       */
049      public SensorProperties() {
050        this(new File(System.getProperty("user.home") + "/.hackystat/v8.sensor.properties"));
051      }
052    
053      /**
054       * Creates a "minimal" sensor properties file usable for test case purposes. Needed by SensorShell
055       * which must be passed a SensorProperties object containing a host, email, and password.
056       *
057       * @param host  The hackystat host.
058       * @param email   The user's email.
059       * @param password   The user's password.
060       */
061      public SensorProperties(String host, String email, String password) {
062        sensorProps.setProperty(SensorProperties.HOST_KEY, host);
063        sensorProps.setProperty(SensorProperties.EMAIL_KEY, email);
064        sensorProps.setProperty(SensorProperties.PASSWORD_KEY, password);
065        // Add the current set of hackystat properties to the System property object.
066        addToSystemProperties(sensorProps);
067        this.fileAvailable = false;
068      }
069    
070      /**
071       * Provides access to Hackystat Sensor settings by reading the passed file.
072       *
073       * @param sensorFile  The sensor file to read.
074       */
075      public SensorProperties(File sensorFile) {
076        this.sensorFile = sensorFile;
077        FileInputStream fileStream = null;
078        try {
079          if (sensorFile.exists()) {
080            fileStream = new FileInputStream(sensorFile);
081            sensorProps.load(fileStream);
082            if (sensorProps.size() > 0) {
083              this.fileAvailable = true;
084              // Add the current set of hackystat properties to the System property object.
085              addToSystemProperties(sensorProps);
086            }
087          }
088        }
089        catch (FileNotFoundException e) {
090          this.fileAvailable = false;
091        }
092        catch (IOException e) {
093          this.fileAvailable = false;
094        }
095        finally {
096          try {
097            fileStream.close();
098          }
099          catch (Exception e) {
100            System.err.println("Error closing stream: " + e);
101          }
102        }
103      }
104    
105    
106    
107      /**
108       * Returns the directory in which the sensor.properties file is located (if it exists). This is
109       * normally the .hackystat directory. If this SensorProperties instance was created without a
110       * sensor.properties file, or if for some other reason the sensor.properties file cannot be found,
111       * then this method returns null.
112       *
113       * @return   A File instance indicating a directory, or null.
114       */
115      public File getSensorPropertiesDir() {
116        if ((sensorFile != null) && (sensorFile.exists())) {
117          return sensorFile.getParentFile();
118        }
119        else {
120          return null;
121        }
122      }
123      
124      /**
125       * Returns the trimmed property value associated with the property key in 
126       * this sensor.properties file.
127       * @param key The parameter key.
128       * @return The trimmed property value associated with this property key, or null if not found.
129       */
130      public String getProperty(String key) {
131        String value = sensorProps.getProperty(key);
132        if (value != null) {
133          value = value.trim();
134        }
135        return value;
136      }
137    
138      /**
139       * Returns the hackystat host. Defaults to http://localhost/.
140       *
141       * @return   The hackystat host.
142       */
143      public String getHackystatHost() {
144        String host = sensorProps.getProperty(HOST_KEY, "http://localhost/").trim(); 
145        if (!host.endsWith("/")) {
146          host = host + "/";
147        }
148        return host;
149      }
150    
151    
152      /**
153       * Returns the absolute path to the properties file, or the empty string if the file is not
154       * available.
155       *
156       * @return   The absolutePath value or the empty string.
157       */
158      public String getAbsolutePath() {
159        return (this.sensorFile == null) ? "" : this.sensorFile.getAbsolutePath();
160      }
161    
162    
163      /**
164       * Returns the password for this user. Defaults to "ChangeThis"
165       *
166       * @return The user password.
167       */
168      public String getPassword() {
169        return sensorProps.getProperty(PASSWORD_KEY, "ChangeThis").trim();
170      }
171    
172      /**
173       * Returns the email for this user. Defaults to "ChangeThis@changethis.com"
174       *
175       * @return The user email.
176       */
177      public String getEmail() {
178        return sensorProps.getProperty(EMAIL_KEY, "ChangeThis@changethis.com").trim();
179      }
180      
181      /**
182       * Returns the AutoSend interval for use by the SensorShell, or 10 if it was not specified. 
183       * Returned as a string since it is typically sent off to the SensorShell as a String argument.
184       *
185       * @return   The autosend interval.
186       */
187      public String getAutoSendInterval() {
188        String intervalString = sensorProps.getProperty(AUTOSEND_KEY, "10").trim();
189        try {
190          // make sure it's an integer.
191          Integer.parseInt(intervalString);
192          return intervalString;
193        }
194        catch (Exception e) {
195          return "10";
196        }
197      }
198    
199      /**
200       * Returns the StateChange interval for use by sensors, or 30 if it was not specified. The
201       * stateChange interval must be greater than 0.
202       *
203       * @return   The state change interval in seconds.
204       */
205      public int getStateChangeInterval() {
206        String intervalString = sensorProps.getProperty(STATECHANGE_KEY, "60").trim();
207        try {
208          int interval = Integer.parseInt(intervalString);
209          return (interval > 0) ? interval : 30;
210        }
211        catch (Exception e) {
212          return 30;
213        }
214      }
215    
216    
217      /**
218       * Returns true if the sensor properties file was found and read successfully.
219       *
220       * @return   True if file was found and readable.
221       */
222      public boolean isFileAvailable() {
223        return this.fileAvailable;
224      }
225    
226      
227      /**
228       * Updates the System properties object with the contents of the passed Hackystat
229       * properties instance.  This method is declared static so that it can be invoked
230       * from both SensorProperties and ServerProperties.  The hackystat properties added from
231       * SensorProperties were found in sensor.properties. The hackystat properties added from
232       * ServerProperties were found in hackystat.server.properties.  
233       * This method is package-private for access by SensorProperties.
234       * @param hackyProperties A Properties instance containing hackystat properties.
235       */
236      private void addToSystemProperties(Properties hackyProperties) {
237        Properties systemProperties = System.getProperties();
238        systemProperties.putAll(hackyProperties);
239        System.setProperties(systemProperties);
240      }
241    }
