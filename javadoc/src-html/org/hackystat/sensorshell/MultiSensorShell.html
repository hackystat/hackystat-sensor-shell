001    package org.hackystat.sensorshell;
002    
003    import java.util.ArrayList;
004    import java.util.Date;
005    import org.hackystat.sensorbase.resource.sensordata.jaxb.Properties;
006    import org.hackystat.sensorbase.resource.sensordata.jaxb.Property;
007    import org.hackystat.sensorbase.resource.sensordata.jaxb.SensorData;
008    import org.hackystat.utilities.tstamp.Tstamp;
009    
010    /**
011     * MultiSensorShell is an experimental class designed to explore performance optimization of 
012     * sensor data transmission.  MultiSensorShell is designed to explore the premise that one way to
013     * improve the throughput of sensor data transmission is to create multiple SensorShells that 
014     * are all sending data to a single SensorBase simultaneously.
015     * <p> 
016     * When instantiating a MultiSensorShell, you specify the number of SensorShells to be created, the
017     * number of SensorData instances to be sent to a single SensorShell before moving on to the next
018     * SensorShell (the "batchSize"), and the autoSendInterval for each SensorShell. Since the 
019     * AutoSend facility in SensorShell creates a separate thread, this enables a simple form of
020     * concurrency: it is possible for one SensorShell to be sending its data while the MultiSensorShell
021     * is adding data to another SensorShell. 
022     * <p>
023     * The main() method provides a simple experimental interface.  To use it, bring up a SensorBase.
024     * Then invoke the main() method, editing the initial local variables as you wish. The system
025     * will run, printing out the elapsed time in milliseconds required for each add() of an 
026     * individual SensorData instance.  At the end of the run, it will print out the total elapsed
027     * time and the average time to send each SensorData instance to the server.  Note that I have
028     * seen a variance of around 0.5 milliseconds in the average time per instance send. So, for
029     * example, when repeatedly running under the same settings, I have obtained values 
030     * varying between 3.1 to about 3.7 milliseconds per instance.  This is with totalData = 5000.
031     * It appears that the variance decreases if you increase totalData to about 50,000. At that 
032     * point, I get relatively consistent throughput of around 2.8 milliseconds per instance. 
033     * <p>
034     * If you instantiate a MultiSensorShell with the number of SensorShells set to 1, you effectively
035     * get the default case.  In this situation, I have found that the average time to send a 
036     * single SensorData instance is approximately 6 milliseconds, almost independent of the 
037     * settings for batchSize and the autoSendInterval.   By increasing the number of SensorShell
038     * instances to 5, I can double the throughput, to approximately 3 milliseconds per instance. At
039     * this point, some kind of performance plateau is reached, and I have not yet been able to 
040     * improve the performance further via manipulation of these variables.  I do not know whether
041     * this is a "real" limit or an artificial limit imposed by my running both client and server
042     * on the same system (a MacBook Pro). 
043     * <p>
044     * My best throughput has been achieved with totalData = 50,000; numShells = 10; autoSendInterval
045     * = 0.05; and batchSize = 200. Under these conditions, I get 2.8 milliseconds per instance, or 
046     * around 360 instances per second throughput.  I've done a run with totalData = 500,000 without
047     * problems and with a sustained throughput of 2.8 milliseconds/instance.   
048     * <p>
049     * I'm also finding I can store around 350,000 sensor data instances per GB of disk space. 
050     * @author Philip Johnson
051     */
052    public class MultiSensorShell {
053      /** The internal SensorShells managed by this MultiSensorShell. */
054      private ArrayList<SensorShell> shells;
055      /** The total number of shells. */
056      private int numShells;
057      /** The number of SensorData instances to be sent to a single Shell before going to the next. */
058      private int batchSize;
059      /** A counter that indicates how many instances have been sent to the current SensorShell. */
060      private int batchCounter = 0;
061      /** A pointer to the current Shell that is receiving SensorData instances. */
062      private int currShellIndex = 0;
063      
064      /**
065       * Creates a new MultiSensorShell for concurrent transmission of SensorData instances to a 
066       * SensorBase. 
067       * @param properties A SensorProperties instance. 
068       * @param autoSendTimeInterval The time in minutes between autoSends for each shell.
069       * @param numShells The total number of SensorShell instances to create. 
070       * @param batchSize The number of SensorData instances to send in a row to a single SensorShell.
071       * @throws Exception If problems occur. 
072       */
073      public MultiSensorShell(SensorProperties properties, double autoSendTimeInterval, int numShells, 
074          int batchSize) 
075      throws Exception {
076        this.shells = new ArrayList<SensorShell>();
077        this.numShells = numShells;
078        this.batchSize = batchSize;
079        for (int i = 0; i < numShells; i++) {
080          SensorShell shell = new SensorShell(properties, false);
081          shell.setAutoSendTimeInterval(autoSendTimeInterval);
082          shell.setAutoSendBufferSize(1000);
083          this.shells.add(shell);
084        }
085      }
086      
087      /**
088       * Adds the passed SensorData instance to one of the internal SensorShells. 
089       * @param sensorData The SensorData instance to be queued for transmission. 
090       */
091      public void add(SensorData sensorData) {
092        // First, update the batchCounter and change the currShellIndex if necessary.
093        // batchCounter goes from 1 to batchSize.
094        // currShellIndex goes from 0 to numShells -1
095        batchCounter++;
096        if (this.batchCounter > batchSize) {
097          batchCounter = 0;
098          this.currShellIndex++;
099          if (this.currShellIndex >= this.numShells) {
100            this.currShellIndex = 0;
101          }
102        }
103        // Now that batchCounter and currShellIndex have the appropriate values, do the add.
104        this.shells.get(currShellIndex).add(sensorData);
105      }
106      
107      /**
108       * Invokes send() on all of the internal SensorShells. Call this at the end of a run to 
109       * ensure that all of the buffered data has been sent. 
110       */
111      public void send() {
112        for (int i = 0; i < numShells; i++) {
113          this.shells.get(i).send();
114        }
115      }
116      
117      /**
118       * Creates a sample SensorData instance given a user. 
119       * @param user The user.
120       * @return The new SensorData instance. 
121       */
122      private static SensorData makeSensorData(String user) {
123        String sdt = "TestSdt";
124        SensorData data = new SensorData();
125        String tool = "Subversion";
126        data.setTool(tool);
127        data.setOwner(user);
128        data.setSensorDataType(sdt);
129        data.setTimestamp(Tstamp.makeTimestamp());
130        data.setResource("file://foo/bar/baz.txt");
131        data.setRuntime(Tstamp.makeTimestamp());
132        Property property = new Property();
133        property.setKey("SampleField");
134        property.setValue("The test value for Sample Field");
135        Properties properties = new Properties();
136        properties.getProperty().add(property);
137        data.setProperties(properties);
138        return data;
139      }
140      
141      /**
142       * A simple main class that illustrates how to invoke the MultiSensorShell to explore 
143       * performance improvements related to throughput of SensorData instances.  The idea is 
144       * bring up a local SensorBase, edit the local variables of this method, then run this 
145       * main() to see how fast you can transmit the SensorData to the SensorBase. 
146       * @param args Ignored.
147       * @throws Exception If things go wrong. 
148       */
149      public static void main(String[] args) throws Exception {
150        long totalData = 50000;
151        int numShells = 10;
152        double autoSendTimeInterval = 0.05;
153        int batchSize = 200;
154        String host = "http://localhost:9876/sensorbase";
155        String user = "TestUser@hackystat.org";
156        SensorProperties properties = new SensorProperties(host, user, user);
157        MultiSensorShell multiShell = 
158          new MultiSensorShell(properties, autoSendTimeInterval, numShells, batchSize);
159        Date startTime = new Date();
160        for (int i = 0; i < totalData; i++) {
161          long time1 = new Date().getTime();
162          SensorData data = makeSensorData(user);
163          multiShell.add(data);
164          long time2 = new Date().getTime();
165          System.out.println("Elapsed millis for add: " + (time2 - time1));
166        }
167        multiShell.send();
168        long totalTime = (new Date().getTime() - startTime.getTime());
169        double timePerData = (double)totalTime / (double)totalData;
170        System.out.println("Total time: " + totalTime + " time/data: " + timePerData);
171    
172      }
173    }
