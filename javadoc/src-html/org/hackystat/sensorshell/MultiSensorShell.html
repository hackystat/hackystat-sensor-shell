001    package org.hackystat.sensorshell;
002    
003    import java.util.ArrayList;
004    import java.util.Map;
005    import org.hackystat.sensorbase.resource.sensordata.jaxb.SensorData;
006    
007    /**
008     * MultiSensorShell is a wrapper around SensorShell that is designed to support high performance
009     * transmission of sensor data instances from a client to a server. Prior research has determined
010     * that when a single SensorShell is used to transmit a large amount of data in a short period of
011     * time, it can spend a substantial portion of its time blocked while waiting for an HTTP PUT to
012     * complete. MultiSensorShell overcomes this problem by instantiating multiple SensorShell instances
013     * internally and then passing sensor data to them in a round-robin fashion. Each SensorShell is
014     * passed an autoSendTimeInterval value, which results in a separate thread for each SensorShell
015     * instance that will concurrently send any buffered data at regular time intervals. This
016     * significantly reduces blocked time for MultiSensorShell, because when any individual SensorShell
017     * instance is performing its HTTP PUT call, the MultiSensorShell can be concurrently adding data to
018     * one of the other SensorShell instances.
019     * <p>
020     * MultiSensorShell provides a number of "tuning parameters", including the number of SensorShells
021     * to be created, the number of SensorData instances to be sent to a single SensorShell before
022     * moving on to the next SensorShell (the "batchSize"), and the autoSendInterval for each
023     * SensorShell. There is a MultiSensorShell constructor that provides reasonable default values for
024     * these tuning parameters and that should result in good performance under most conditions.
025     * <p>
026     * The TestMultiSensorShell class provides a main() method that we have used to do some simple
027     * performance evaluation, which we report on next. All results were obtained using a MacBook Pro
028     * with a 2.33 Ghz Intel Core Duo processor and 3 GB of 667 Mhz DDR2 SDRAM. Both the client and
029     * SensorBase server were running on this computer to minimize network latency issues.
030     * <p>
031     * If you instantiate a MultiSensorShell with the number of SensorShells set to 1, you effectively
032     * get the default case. In this situation, we have found that the average time to send a single
033     * SensorData instance is approximately 6 milliseconds, almost independent of the settings for
034     * batchSize and the autoSendInterval. Increasing the number of SensorShell instances to 5 doubles
035     * the throughput, to approximately 3 milliseconds per instance. At this point, some kind of
036     * performance plateau is reached, with further tweaking of the tuning parameters seeming to have
037     * little effect. We do not know whether this is a "real" limit or an artificial limit based upon
038     * some environmental feature (such as the logging mechanism.)
039     * <p>
040     * With totalData = 500,000, numShells = 10, autoSendInterval = 0.05, and batchSize = 200, we have
041     * achieved throughput of 2.8 milliseconds per instance (which is equivalent to 360 instances per
042     * second and 1.2M instances per hour.)
043     * <p>
044     * We have also found that we can store around 350,000 sensor data instances per GB of disk space.
045     * <p>
046     * Note that a single SensorShell instance is simpler, creates less processing overhead, and has
047     * equivalent performance to MultiSensorShell for transmission loads up to a dozen or so sensor data
048     * instances per second. We recommend using a single SensorShell instance rather than
049     * MultiSensorShell unless optimizing data transmission throughput is an important requirement.
050     * 
051     * @author Philip Johnson
052     */
053    public class MultiSensorShell {
054      /** The internal SensorShells managed by this MultiSensorShell. */
055      private ArrayList<SensorShell> shells;
056      /** The total number of shells. */
057      private int numShells;
058      /** The number of SensorData instances to be sent to a single Shell before going to the next. */
059      private int batchSize;
060      /** A counter that indicates how many instances have been sent to the current SensorShell. */
061      private int batchCounter = 0;
062      /** A pointer to the current Shell that is receiving SensorData instances. */
063      private int currShellIndex = 0;
064    
065      /**
066       * Creates a new MultiSensorShell for multi-threaded transmission of SensorData instances to a
067       * SensorBase.
068       * 
069       * @param properties A SensorProperties instance.
070       * @param toolName The name of the tool, used to name the log file.
071       * @param enableOfflineData Whether or not to save/restore offline data.
072       * @param autoSendTimeInterval The time in minutes between autoSends for each shell.
073       * @param autoSendBatchSize The maximum buffer size for each shell.
074       * @param numShells The total number of SensorShell instances to create. Must be greater than 0.
075       * @param batchSize The number of SensorData instances to send in a row to a single SensorShell.
076       * Must be greater than 0.
077       * @throws Exception If numShells or batchSize values are illegal, or other problems occur.
078       */
079      public MultiSensorShell(SensorProperties properties, String toolName, boolean enableOfflineData,
080          double autoSendTimeInterval, int autoSendBatchSize, int numShells, int batchSize)
081          throws Exception {
082        if (numShells < 1) {
083          throw new Exception("Number of shells must be greater than 0.");
084        }
085        if (batchSize < 1) {
086          throw new Exception("Batch size must be greater than 0.");
087        }
088        this.shells = new ArrayList<SensorShell>();
089        this.numShells = numShells;
090        this.batchSize = batchSize;
091        for (int i = 0; i < numShells; i++) {
092          // MultiSensorShells must always be non-interactive.
093          boolean isInteractive = false;
094          SensorShell shell = new SensorShell(properties, isInteractive, toolName, enableOfflineData);
095          shell.setAutoSendTimeInterval(autoSendTimeInterval);
096          shell.setAutoSendBufferSize(autoSendBatchSize);
097          this.shells.add(shell);
098        }
099      }
100    
101      /**
102       * Constructs a new MultiSensorShell instance with reasonable default values for enableOfflineData
103       * (true), autoSendTimeInterval (6 seconds), autoSendBatchSize (1000), numShells (5), and
104       * batchSize (200).
105       * 
106       * @param sensorProperties The sensor properties instance for this run.
107       * @param toolName Indicates the invoking tool that is added to the log file name.
108       * @throws Exception If problems occur.
109       */
110      public MultiSensorShell(SensorProperties sensorProperties, String toolName) throws Exception {
111        this(sensorProperties, toolName, true, 0.1, 1000, 5, 200);
112      }
113    
114      /**
115       * Adds the passed SensorData instance to one of the internal SensorShells.
116       * 
117       * @param sensorData The SensorData instance to be queued for transmission.
118       */
119      public void add(SensorData sensorData) {
120        this.shells.get(getCurrShellIndex()).add(sensorData);
121      }
122    
123      /**
124       * Converts the values in the KeyValMap to a SensorData instance and adds it to one of the
125       * internal SensorShells. Owner will default to the hackystat user in the sensor.properties file.
126       * Timestamp and Runtime will default to the current time.
127       * 
128       * @param keyValMap The map of key-value pairs.
129       * @throws Exception If the Map cannot be translated into SensorData, typically because a value
130       * was passed for Timestamp or Runtime that could not be parsed into XMLGregorianCalendar.
131       */
132      public void add(Map<String, String> keyValMap) throws Exception {
133        this.shells.get(getCurrShellIndex()).add(keyValMap);
134      }
135    
136      /**
137       * Returns an index to the current SensorShell index to be used for data transmission. Internally
138       * updates the batchCounter.
139       * 
140       * @return The index to the current SensorShell instance.
141       */
142      private int getCurrShellIndex() {
143        // First, update the batchCounter and change the currShellIndex if necessary.
144        // batchCounter goes from 1 to batchSize.
145        // currShellIndex goes from 0 to numShells -1
146        batchCounter++;
147        if (this.batchCounter > batchSize) {
148          batchCounter = 0;
149          this.currShellIndex++;
150          if (this.currShellIndex >= this.numShells) {
151            this.currShellIndex = 0;
152          }
153        }
154        return currShellIndex;
155      }
156    
157      /**
158       * Returns true if the host can be pinged and the email/password combination is valid.
159       * 
160       * @return True if the host can be pinged and the user credentials are valid.
161       */
162      public boolean ping() {
163        return this.shells.get(0).ping();
164      }
165    
166      /**
167       * Immediately invokes send() on all of the internal SensorShells. Note that you will rarely want
168       * to invoke this method. Instead, during normal operation you will rely on the
169       * autoSendTimeInterval to invoke send() in a separate thread, and then invoke quit() to invoke
170       * send() at the conclusion of the run.
171       * @return The total number of instances sent by all internal SensorShells. 
172       */
173      public int send() {
174        int totalSent = 0;
175        for (int i = 0; i < numShells; i++) {
176          totalSent += this.shells.get(i).send();
177        }
178        return totalSent;
179      }
180    
181      /**
182       * Invokes quit() on all of the internal SensorShells, which invokes a final send() and closes any
183       * logging files.
184       */
185      public void quit() {
186        for (int i = 0; i < numShells; i++) {
187          this.shells.get(i).quit();
188        }
189      }
190    }
