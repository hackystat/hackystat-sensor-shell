001    package org.hackystat.sensorshell;
002    
003    import java.util.ArrayList;
004    import java.util.Map;
005    import java.util.Random;
006    import java.util.logging.Level;
007    
008    import org.hackystat.sensorbase.resource.sensordata.jaxb.SensorData;
009    
010    /**
011     * MultiSensorShell is a wrapper around SensorShell that is designed to support high performance
012     * transmission of sensor data instances from a client to a server. Prior research has determined
013     * that when a single SensorShell is used to transmit a large amount of data in a short period of
014     * time, it can spend a substantial portion of its time blocked while waiting for an HTTP PUT to
015     * complete. MultiSensorShell overcomes this problem by instantiating multiple SensorShell instances
016     * internally and then passing sensor data to them in a round-robin fashion. Each SensorShell is
017     * passed an autoSendTimeInterval value, which results in a separate thread for each SensorShell
018     * instance that will concurrently send any buffered data at regular time intervals. This
019     * significantly reduces blocked time for MultiSensorShell, because when any individual SensorShell
020     * instance is performing its HTTP PUT call, the MultiSensorShell can be concurrently adding data to
021     * one of the other SensorShell instances.
022     * <p>
023     * MultiSensorShell provides a number of "tuning parameters", including the number of SensorShells
024     * to be created, the number of SensorData instances to be sent to a single SensorShell before
025     * moving on to the next SensorShell (the "batchSize"), and the autoSendInterval for each
026     * SensorShell. There is a MultiSensorShell constructor that provides reasonable default values for
027     * these tuning parameters and that should result in good performance under most conditions.
028     * <p>
029     * The TestMultiSensorShell class provides a main() method that we have used to do some simple
030     * performance evaluation, which we report on next. All results were obtained using a MacBook Pro
031     * with a 2.33 Ghz Intel Core Duo processor and 3 GB of 667 Mhz DDR2 SDRAM. Both the client and
032     * SensorBase server were running on this computer to minimize network latency issues.
033     * <p>
034     * If you instantiate a MultiSensorShell with the number of SensorShells set to 1, you effectively
035     * get the default case. In this situation, we have found that the average time to send a single
036     * SensorData instance is approximately 6 milliseconds, almost independent of the settings for
037     * batchSize and the autoSendInterval. Increasing the number of SensorShell instances to 5 doubles
038     * the throughput, to approximately 3 milliseconds per instance. At this point, some kind of
039     * performance plateau is reached, with further tweaking of the tuning parameters seeming to have
040     * little effect. We do not know whether this is a "real" limit or an artificial limit based upon
041     * some environmental feature (such as the logging mechanism.)
042     * <p>
043     * With totalData = 500,000, numShells = 10, autoSendInterval = 0.05, and batchSize = 200, we have
044     * achieved throughput of 2.8 milliseconds per instance (which is equivalent to 360 instances per
045     * second and 1.2M instances per hour.)
046     * <p>
047     * We have also found that we can store around 350,000 sensor data instances per GB of disk space.
048     * <p>
049     * Note that although autoSendBatchSize is provided as a tuning parameter, you will probably want
050     * to set this to a high value (20,000 or 30,000) to effectively disable it. This is because 
051     * reaching the batchSize limit forces to blocking send() of the data, which is precisely what we
052     * want to avoid in MultiSensorShell.  Instead, we want to tune the autoSendTimeInterval so that all
053     * of our send() invocations occur asynchronously in a separate thread. 
054     * <p>
055     * Note that a single SensorShell instance is simpler, creates less processing overhead, and has
056     * equivalent performance to MultiSensorShell for transmission loads up to a dozen or so sensor data
057     * instances per second. We recommend using a single SensorShell instance rather than
058     * MultiSensorShell unless optimizing data transmission throughput is an important requirement.
059     * 
060     * @author Philip Johnson
061     */
062    public class MultiSensorShell implements Shell {
063      /** The internal SensorShells managed by this MultiSensorShell. */
064      private ArrayList<SensorShell> shells;
065      /** The total number of shells. */
066      private int numShells;
067      /** The number of SensorData instances to be sent to a single Shell before going to the next. */
068      private int batchSize;
069      /** A counter that indicates how many instances have been sent to the current SensorShell. */
070      private int batchCounter = 0;
071      /** A pointer to the current Shell that is receiving SensorData instances. */
072      private int currShellIndex = 0;
073      /** Used when batchSize == 0 */
074      private Random generator = new Random(0L);
075    
076      /**
077       * Creates a new MultiSensorShell for multi-threaded transmission of SensorData instances to a
078       * SensorBase.
079       * 
080       * @param properties A SensorProperties instance.
081       * @param toolName The name of the tool, used to name the log file.
082       * @param enableOfflineData Whether or not to save/restore offline data.
083       * @param autoSendTimeInterval The time in minutes between autoSends for each shell.
084       * @param autoSendBatchSize The maximum buffer size for each shell.
085       * @param numShells The total number of SensorShell instances to create. Must be greater than 0.
086       * @param batchSize The number of SensorData instances to send in a row to a single SensorShell.
087       * Must be greater than or equal to 0. If equal to 0, each instance is sent to a shell picked at 
088       * random.
089       * @throws Exception If numShells or batchSize values are illegal, or other problems occur.
090       */
091      public MultiSensorShell(SensorProperties properties, String toolName, boolean enableOfflineData,
092          double autoSendTimeInterval, int autoSendBatchSize, int numShells, int batchSize)
093          throws Exception {
094        if (numShells < 1) {
095          throw new Exception("Number of shells must be greater than 0.");
096        }
097        if (batchSize < 0) {
098          throw new Exception("Batch size must be greater than or equal to 0.");
099        }
100        this.shells = new ArrayList<SensorShell>();
101        this.numShells = numShells;
102        this.batchSize = batchSize;
103        for (int i = 0; i < numShells; i++) {
104          // MultiSensorShells must always be non-interactive.
105          boolean isInteractive = false;
106          SensorShell shell = new SensorShell(properties, isInteractive, toolName, enableOfflineData);
107          shell.setAutoSendTimeInterval(autoSendTimeInterval);
108          //autoSendTimeInterval += 0.001; // see if slightly changing send times helps. It doesn't.
109          shell.setAutoSendBufferSize(autoSendBatchSize);
110          this.shells.add(shell);
111        }
112      }
113    
114      /**
115       * Constructs a new MultiSensorShell instance with reasonable default values for enableOfflineData
116       * (true), autoSendTimeInterval (6 seconds), autoSendBatchSize (1000), numShells (10), and
117       * batchSize (1000).
118       * 
119       * @param sensorProperties The sensor properties instance for this run.
120       * @param toolName Indicates the invoking tool that is added to the log file name.
121       * @throws Exception If problems occur.
122       */
123      public MultiSensorShell(SensorProperties sensorProperties, String toolName) throws Exception {
124        this(sensorProperties, toolName, true, 0.1, 1000, 10, 200);
125      }
126    
127      /** {@inheritDoc} */
128      public void add(SensorData sensorData) {
129        this.shells.get(getCurrShellIndex()).add(sensorData);
130      }
131    
132      /** {@inheritDoc} */
133      public void add(Map<String, String> keyValMap) throws Exception {
134        this.shells.get(getCurrShellIndex()).add(keyValMap);
135      }
136    
137      /**
138       * Returns an index to the current SensorShell index to be used for data transmission. Internally
139       * updates the batchCounter.
140       * If batchSize is 0, then an index is returned at random. In our initial trials, this was found
141       * to be a suboptimal strategy; it is better to set the batchSize to something like 200.
142       * 
143       * @return The index to the current SensorShell instance.
144       */
145      private int getCurrShellIndex() {
146        // If batchSize is 0, then we return a shell index chosen randomly.
147        if (batchSize == 0) {
148          return generator.nextInt(numShells);
149        }
150        // Now, update the batchCounter and change the currShellIndex if necessary.
151        // batchCounter goes from 1 to batchSize.
152        // currShellIndex goes from 0 to numShells -1
153        batchCounter++;
154        if (this.batchCounter > batchSize) {
155          batchCounter = 0;
156          this.currShellIndex++;
157          if (this.currShellIndex >= this.numShells) {
158            this.currShellIndex = 0;
159          }
160        }
161        return currShellIndex;
162      }
163    
164      /**
165       * Returns true if the host can be pinged and the email/password combination is valid.
166       * 
167       * @return True if the host can be pinged and the user credentials are valid.
168       */
169      public boolean ping() {
170        return this.shells.get(0).ping();
171      }
172    
173      /** {@inheritDoc} */
174      public int send() {
175        int totalSent = 0;
176        for (int i = 0; i < numShells; i++) {
177          totalSent += this.shells.get(i).send();
178        }
179        return totalSent;
180      }
181    
182      /** {@inheritDoc} */
183      public void quit() {
184        for (int i = 0; i < numShells; i++) {
185          this.shells.get(i).quit();
186        }
187      }
188      
189      /** {@inheritDoc} */
190      public void setLoggingLevel(String level) {
191        Level newLevel = Level.INFO;
192        try {
193          newLevel = Level.parse(level);
194        }
195        catch (Exception e) {
196          this.shells.get(0).getLogger().warning("Couldn't set Logging level to: " + level);
197        }
198        for (int i = 0; i < numShells; i++) {
199          this.shells.get(i).getLogger().setLevel(newLevel);
200          this.shells.get(i).getLogger().getHandlers()[0].setLevel(newLevel);
201        }
202      }
203    }
