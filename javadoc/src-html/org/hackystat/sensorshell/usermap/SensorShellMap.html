001    package org.hackystat.sensorshell.usermap;
002    
003    import java.io.File;
004    import java.util.HashMap;
005    
006    import org.hackystat.sensorshell.SensorProperties;
007    import org.hackystat.sensorshell.SensorShell;
008    import org.hackystat.sensorshell.usermap.UserMap.UserMapKey;
009    
010    /**
011     * Allows for lazy instantiation of SensorShells for user tool accounts. This class abstracts the
012     * use of the UserMap class so that sensors for applicable tool sensors won't have to manage it.
013     * <p>
014     * The <code>SensorShellMap</code> is meant to only derive SensorShells for a specific tool.
015     * Queries for SensorShells are done by giving an account name to the <code>getUserShell</code>
016     * method. This account name must be specific to the tool of the <code>SensorShellMap</code>
017     * instance.
018     * <p>
019     * Note that the "tool" name is used in a case-insensitive fashion.
020     * 
021     * @author Burt Leung, Julie Ann Sakuda (v8 port)
022     */
023    public class SensorShellMap {
024    
025      /** The tool that this SensorShellMap will return SensorShells for. */
026      private String tool;
027    
028      /** Mapping of Tool specific user accounts to their Hackystat information. */
029      private UserMap userMap;
030    
031      /** Map of Tool specific tool accounts to their SensorShells. */
032      private HashMap<String, SensorShell> toolAccountsToShells = new HashMap<String, SensorShell>();
033    
034      /**
035       * Instantiate this class which initializes the UserMap used to get SensorShells for users and
036       * also sets the Hackystat host for the SensorShells.
037       * 
038       * @param tool The specific tool that this SensorShellMap will provide SensorShells for.
039       * @throws SensorShellMapException Indicates that the usermaps.xml file is not found.
040       */
041      public SensorShellMap(String tool) throws SensorShellMapException {
042        // Ensure valid argument
043        if (tool == null || tool.length() == 0) {
044          throw new SensorShellMapException("Error: tool is null or empty string.");
045        }
046        this.tool = tool.trim();
047        this.userMap = new UserMap();
048      }
049    
050      /**
051       * A package private constructor meant to be used only by Junit test cases.
052       * 
053       * @param tool The specific tool that this SensorShellMap will provide SensorShells for.
054       * @param userMapFile The UserMap.xml file.
055       * @throws SensorShellMapException Indicates that the UserMap.xml file is not found.
056       */
057      public SensorShellMap(String tool, File userMapFile)
058          throws SensorShellMapException {
059        this.tool = tool;
060        this.userMap = new UserMap(userMapFile);
061      }
062    
063      /**
064       * Returns true if toolAccount is known so that a corresponding SensorShell can be retrieved.
065       * 
066       * @param toolAccount The toolAccount for the tool in this instance, such as "johnson".
067       * @return True if toolAccount is known, otherwise false.
068       */
069      public boolean hasUserShell(String toolAccount) {
070        if (toolAccount == null || toolAccount.length() == 0) {
071          return false;
072        }
073        else {
074          return this.userMap.hasUser(this.tool, toolAccount.trim());
075        }
076      }
077    
078      /**
079       * Gets the SensorShell instance for a Hackystat user with an account for the given tool. Assumes
080       * that toolAccount is known and will map to a userKey; use hasUserShell() to check whether this
081       * is true before calling this method. Instantiates the SensorShell instance if it is not yet
082       * available.
083       * 
084       * @param toolAccount The name of the account for the given tool. This is not the same as the
085       *          Hackystat account name although it may be the same.
086       * @return The SensorShell instance for the specific Hackystat user.
087       * @throws SensorShellMapException When the user account and/or tool is undefined.
088       */
089      public SensorShell getUserShell(String toolAccount) throws SensorShellMapException {
090        // Check argument that it is valid
091        if (toolAccount == null || toolAccount.length() == 0) {
092          throw new SensorShellMapException("Error: toolAccount is null or the empty string.");
093        }
094        String trimmedToolAccount = toolAccount.trim();
095        try {
096          String user = this.userMap.get(this.tool, trimmedToolAccount, UserMapKey.USER);
097          String password = this.userMap.get(this.tool, trimmedToolAccount, UserMapKey.PASSWORD);
098          String sensorbase = this.userMap.get(this.tool, trimmedToolAccount, UserMapKey.SENSORBASE);
099          // If we haven't instantiated this shell yet, then do it now.
100          if (this.toolAccountsToShells.get(toolAccount) == null) {
101            SensorShell userShell = new SensorShell(new SensorProperties(sensorbase, user, password),
102                false, this.tool);
103            this.toolAccountsToShells.put(trimmedToolAccount, userShell);
104          }
105          // Now return the SensorShell.
106          return this.toolAccountsToShells.get(trimmedToolAccount);
107        }
108        catch (Exception e) {
109          throw new SensorShellMapException("Error getting SensorShell for " + toolAccount, e);
110        }
111      }
112    }
