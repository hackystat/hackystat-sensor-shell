001    package org.hackystat.sensorshell.usermap;
002    
003    import java.io.File;
004    
005    import org.hackystat.sensorshell.usermap.UserMap.UserMapKey;
006    
007    import junit.framework.TestCase;
008    
009    /**
010     * A test case class to test the UserMap class. It makes sure that instantiation, and therefore
011     * formatting of the UserMap.xml file, is done correctly. It also ensures that it correctly provides
012     * its core functionality by testing to see that it gives back a dummy Hackystat value for a given
013     * dummy tool account.
014     * 
015     * @author Julie Ann Sakuda
016     */
017    public class TestUserMap extends TestCase {
018    
019      private String tool = "test";
020      private String toolAccount = "dummyaccount";
021      private String user = "test@hackystat.org";
022      private String password = "test@hackystat.org";
023      private String sensorbase = "http://localhost/";
024    
025      /**
026       * Test the UserMap class and the formatting of the UserMap.xml.
027       * 
028       * @throws Exception if errors occur.
029       */
030      public void testMap() throws Exception {
031        UserMap userMap = null;
032        userMap = new UserMap(new File(System.getProperty("usermaptestfile")));
033        assertNotNull("Checking that the UserMap is not null", userMap);
034        assertEquals("Check user", user, userMap.get(tool, toolAccount, UserMapKey.USER));
035        assertEquals("Check password", password, userMap.get(tool, toolAccount, UserMapKey.PASSWORD));
036        assertEquals("Check sensorbase", sensorbase, userMap.get(tool, toolAccount,
037            UserMapKey.SENSORBASE));
038      }
039    }
