026    /**
027     * Provides "middleware" for accumulating and sending notification (sensor)
028     * data to Hackystat. SensorShell has two modes of interaction: command line and
029     * programmatic. 
030     * <p> 
031     * Command line mode is entered by invoking the main() method, and
032     * is intended to be used as a kind of subshell to which commands to add and
033     * send notification data of various types can be sent. The SensorShell can be invoked
034     * without any additional arguments as follows:
035     * 
036     * <pre>java -jar sensorshell.jar</pre>
037     * 
038     * Or you can invoke it with one, two, three, or four additional arguments:
039     * 
040     * <pre>java -jar sensorshell.jar [toolname] [sensor.properties] [no offline] [command file]</pre>
041     * <p>
042     * Programmatic mode involves creating an instance of SensorShell, retrieving the 
043     * appropriate command instance (Ping, Add, etc.) and invoking the appropriate method.
044     *
045     * @author    Philip M. Johnson
046     */

130      /**
131       * Constructs a new SensorShell instance that can be provided with
132       * notification data to be sent eventually to a specific user key and host.
133       * (For testing purposes only, you may want to disable offline data recovery.)
134       *
135       * @param sensorProperties   The sensor properties instance for this run.
136       * @param isInteractive      Whether this SensorShell is being interactively invoked or not.
137       * @param toolName           The invoking tool that is added to the log file name.
138       * @param enableOfflineData  A boolean indicating whether to recover and/or store offline data.
139       * @param commandFile        A file containing shell commands, or null if none provided.
140       */
141      public SensorShell(SensorProperties sensorProperties, boolean isInteractive, String toolName,
142          boolean enableOfflineData, File commandFile) {
143        this.isInteractive = isInteractive;
144        this.toolName = toolName;
145        this.sensorProperties = sensorProperties;
146        this.enableOfflineData = enableOfflineData;
147        this.commandFile = commandFile;
148        this.commandFilePresent = ((commandFile != null));
149        this.client = new SensorBaseClient(sensorProperties.getHackystatHost(), 
150            sensorProperties.getEmail(), sensorProperties.getPassword());
151        // Create the offline directory in the same directory where sensor.properties was found.
152        //OfflineManager.getInstance(sensorProperties.getSensorPropertiesDir());
153        initializeLogger();
154        this.pingCommand = new PingCommand(this, sensorProperties);
155        this.sensorDataCommand = new SensorDataCommand(this, sensorProperties, this.pingCommand, 
156            this.client);
157        this.autoSendCommand = new AutoSendCommand(this, sensorProperties);
158        this.quitCommand = new QuitCommand(this, sensorProperties, this.sensorDataCommand);
159        printBanner(sensorProperties);
160        this.autoSendCommand.initialize();
161    
162        // Now determine whether to read commands from the input stream or from the command file.
163        try {
164          this.bufferedReader = (commandFilePresent) ?
165            new BufferedReader(new FileReader(this.commandFile)) :
166            new BufferedReader(new InputStreamReader(System.in));
167        }
168        catch (IOException e) {
169          this.logger.info(cr);
170        }
171        this.offlineManager = new OfflineManager(this);
172        recoverOfflineData();
173      }

200      /**
201       * Looks for offline data and recovers any that is found if offline data
202       * management is enabled and if the server is currently pingable.
203       */
204      private void recoverOfflineData() {
205        // Return immediately if server is not available.
206        if (!this.pingCommand.isPingable()) {
207          this.println("Server not available; offline data not recovered.");
208          return;
209        }
210        // Return immediately if the SensorShell was instantiated with offline data disabled.
211        if (!this.enableOfflineData) {
212          this.println("Offline data management disabled.");
213          return;
214        }
215        // Otherwise do offline data management.
216        this.println("Checking for offline data to recover.");
217        this.offlineManager.recover();
218      }

221      /**
222       * Initializes SensorShell logging. All client input is recorded to a log file
223       * in .hackystat/logs/.
224       */
225      private void initializeLogger() {
226        try {
227          // First, create the logs directory, using the same location as sensor.properties if possible.
228          File logDir = (sensorProperties.getSensorPropertiesDir() == null) ?
229              new File(System.getProperty("user.home") + "/.hackystat/logs") :
230              new File(sensorProperties.getSensorPropertiesDir(), "/logs");
231          logDir.mkdirs();
232    
233          // Now set up logging to a file in that directory.
234          this.logger = Logger.getLogger("hackystat.client.cli.SensorShell");
235          this.logger.setUseParentHandlers(false);
236          String fileName = logDir.getAbsolutePath() + "/" + this.toolName + ".%u.log";
237          FileHandler handler = new FileHandler(fileName, 500000, 1, true);
238          handler.setFormatter(this.oneLineFormatter);
239          this.logger.addHandler(handler);
240          // Add a couple of newlines to the log file to distinguish new shell sessions.
241          logger.info(cr + cr);
242        }
243        catch (Exception e) {
244          System.out.println("Error initializing SensorShell logger:\n" + e);
245        }
246      }

249       * Sets the logging level to be used for this logger.

253      public synchronized void setLoggingLevel(String level) {
254        Level newLevel = Level.INFO;
255        try {
256          newLevel = Level.parse(level);
257        }
258        catch (Exception e) {
259          logger.info("Couldn't set Logging level to: " + level);
260        }
261        logger.setLevel(newLevel);
262        logger.getHandlers()[0].setLevel(newLevel);
263      }

288      /**
289       * Process a single input string representing a command.
290       * @param inputString A command as a String.
291       */
292      private void processInputString(String inputString) {
293        // Ignore empty commands. 
294        if ((inputString == null) || ("".equals(inputString))) {
295          return;
296        }
297        // Log the command if we're not running interactively.
298        if (!this.isInteractive) {
299          logger.info("#> " + inputString);
300        }
301        // Process quit command.
302        if ("quit".equals(inputString)) {
303          this.quitCommand.quit();
304          return;
305        }
306    
307        // Process help command.
308        if ("help".equals(inputString)) {
309          this.printHelp();
310          return;
311        }
312    
313        // Process send command.
314        if ("send".equals(inputString)) {
315          this.sensorDataCommand.send();
316          return;
317        }
318        
319        // Process ping command.
320        if ("ping".equals(inputString)) {
321          boolean isPingable = this.pingCommand.isPingable();
322          this.println("Ping of host " + this.sensorProperties.getHackystatHost() + " for user " +
323              this.sensorProperties.getEmail() + 
324              (isPingable ? " succeeded." : " did not succeed"));
325          return;
326        }
327    
328        // Process commands with arguments. 
329        StringTokenizer tokenizer = new StringTokenizer(inputString, this.delimiter);
330        int numTokens = tokenizer.countTokens();
331        // All remaining commands must have arguments. 
332        if (numTokens == 0) {
333          this.println("Error: unknown command or command requires arguments.");
334          return;
335        }
336    
337        // Get the command name and any arguments. 
338        String commandName = tokenizer.nextToken();
339        ArrayList<String> argList = new ArrayList<String>();
340        while (tokenizer.hasMoreElements()) {
341          argList.add(tokenizer.nextToken());
342        }
343        
344        if ("add".equals(commandName)) {
345          // For an Add command, the argument list should be a set of key-value pairs.
346          // So, build the Map of key-value pairs.
347          Map<String, String> keyValMap = new HashMap<String, String>();
348          for (String arg : argList) {
349            int delim = arg.indexOf('=');
350            if (delim == -1) {
351              this.println("Error: can't parse argument string for add command.");
352            }
353            keyValMap.put(arg.substring(0, delim), arg.substring(delim + 1));
354          }
355          try {
356            this.sensorDataCommand.add(keyValMap);
357          }
358          catch (Exception e) {
359            this.println("Error: Can't parse the Timestamp or Runtime arguments.");
360          }
361          return;
362        }
363        
364        if ("statechange".equals(commandName)) {
365          String resourceCheckSumString = argList.get(0);
366          int resourceCheckSum = 0;
367          try {
368            resourceCheckSum = Integer.parseInt(resourceCheckSumString);
369          }
370          catch (Exception e) {
371            this.println("Error: Can't parse the checksum into an integer.");
372            return;
373          }
374          argList.remove(0);
375          // Now do almost the same as for an add command. 
376          // Build the Map of key-value pairs.
377          Map<String, String> keyValMap = new HashMap<String, String>();
378          for (String arg : argList) {
379            int delim = arg.indexOf('=');
380            if (delim == -1) {
381              this.println("Error: can't parse argument string for statechange command.");
382            }
383            keyValMap.put(arg.substring(0, delim), arg.substring(delim + 1));
384          }
385          try {
386            this.sensorDataCommand.statechange(resourceCheckSum, keyValMap);
387          }
388          catch (Exception e) {
389            this.println("Error: Can't parse the Timestamp or Runtime arguments.");
390          }
391          return;
392        }
393        
394        if ("setAutoSendTimeInterval".equals(commandName)) {
395          String interval = (argList.isEmpty()) ? "" : argList.get(0);
396          this.autoSendCommand.setTimeInterval(interval);
397          return;
398        }
399        
400        if ("setAutoSendBufferSize".equals(commandName)) {
401          String bufferSize = (argList.isEmpty()) ? "" : argList.get(0);
402          this.autoSendCommand.setBufferSize(bufferSize);
403          return;
404        }
405    
406        // Otherwise we don't understand.
407        this.println("Invalid command entered and ignored. Type 'help' for help.");
408      }

411      /**
412       * The command line shell interface.
413       * <ul>
414       *   <li> If invoked with no arguments, then the default sensor.properties
415       *   file is used and the toolName field in the sensor log file name is
416       *   "interactive".
417       *   <li> If invoked with one argument, that argument is used as the toolName
418       *   value.
419       *   <li> If invoked with two arguments, then the first argument is used as
420       *   the toolName value and the second is used as the sensor.properties file
421       *   path.
422       *   <li> If invoked with three arguments, then the first argument is the
423       *   toolname, the second is the sensor.properties file path, and the third
424       *   argument (regardless of value) disables offline data management.
425       *   <li> If invoked with four arguments, then the first argument is the
426       *   toolname, the second is the sensor.properties file path, the third
427       *   argument (regardless of value) disables offline data management, and
428       *   the fourth argument specifies a file of commands (the last of which should be 'quit').
429       * </ul>
430       * Unless four arguments are provided,
431       * the shell then provides a ">>" prompt and supports interactive entry of
432       * sensor data. The following commands are supported:
433       * <ul>
434       *   <li> "help" provides a summary of the available commands.
435       *   <li> "send" sends all of the accumulated data to the server.
436       *   <li> "setAutoSendTimeInterval" specifies the number of minutes between auto sending.
437       *   <li> "setAutoSendBufferSize" specifies the maximum number of local entries before sending.
438       *   <li> "quit" sends all of the accumulated data to the server and exits.
439       *   <li> "ping" checks to see if the host/user/password is valid. 
440       *   <li> "add" adds a single Sensor Data instance to the buffered list to send.
441       * </ul>
442       *
443       * @param args  The command line parameters. See above for details.
444       */
445      public static void main(String args[]) {
446        // Print help line and exit if arg is -help.
447        if ((args.length == 1) && (args[0].equalsIgnoreCase("-help"))) {
448          System.out.println("java -jar sensorshell.jar [toolname] [sensor.properties] [no offline] "
449               + "[command filename]");
450          return;
451        }
452        
453        // Set Parameter 1 (toolname) to supplied or default value.
454        String toolName = (args.length > 0) ? args[0] : "interactive";
455    
456        // Set Parameter 2 (sensor properties file) to supplied or default value. Exit if can't find it.
457        SensorProperties sensorProperties = null;
458        try {
459          sensorProperties = (args.length >= 2) ?
460              new SensorProperties(new File(args[1])) : new SensorProperties();
461        }
462        catch (SensorPropertiesException e) {
463          System.out.println(e.getMessage());
464          System.out.println("Exiting...");
465          return;
466        }
467    
468        // Set Parameter 3 (offline). True if we don't supply a value, false if any value supplied.
469        boolean offlineEnabled = ((args.length < 3));
470    
471        // Set Parameter 4 (command file). Null if not supplied. Exit if supplied and bogus.
472        File commandFile = null;
473        if (args.length == 4) {
474          commandFile = new File(args[3]);
475          if (!(commandFile.exists() && commandFile.isFile())) {
476            System.out.println("Could not find the command file. Exiting...");
477            return;
478          }
479        }
480    
481        // Set interactive parameter. From command line, always interactive unless using command file.
482        boolean interactive = ((commandFile == null));
483    
484        // Now create the shell instance, supplying it with all the appropriate arguments.
485        SensorShell shell = 
486          new SensorShell(sensorProperties, interactive, toolName, offlineEnabled, commandFile);
487    
488        // Start processing commands either interactively or from the command file.
489        while (true) {
490          // Get the next command
491          shell.printPrompt();
492          String inputString = shell.readLine();
493          shell.processInputString(inputString);
494          if (inputString.equalsIgnoreCase("quit")) {
495            return;
496          }
497        }
498      }

500      /** Prints the help strings associated with all commands. */
501      private void printHelp() {
502        String helpString = 
503          "SensorShell Command Summary " + cr
504          + "  add#<key>=<value>[#<key>=<value>]..." + cr
505          + "    Adds a new sensor data instance for subsequent sending." + cr 
506          + "    Provide fields and properties as key=value pairs separated by '#'." + cr
507          + "    Owner, Timestamp, and Runtime fields will default to the current user and time." + cr
508          + "    Example: add#Tool=Eclipse#SensorDataType=DevEvent#DevEventType=Compile" + cr 
509          + "  send" + cr
510          + "    Sends any added sensor data to the server. " + cr
511          + "    Server is pinged, and if it does not respond, then data is stored offline." + cr
512          + "    Example: send" + cr
513          + "  ping" + cr
514          + "    Pings the server and checks email/password credentials." + cr
515          + "    Example: ping" + cr
516          + "  setAutoSendTimeInterval#<integer>" + cr
517          + "    Sets the interval in minutes between automatic sending of added sensor data." + cr
518          + "    Provide 0 as the integer to disable autosend. Fractional values to 0.1 minutes OK" + cr
519          + "    Example: setAutoSendTimeInterval#15" + cr
520          + "  setAutoSendBufferSize#<integer>" + cr
521          + "    Sets the maximum size of the local buffer before sending" + cr
522          + "    Example: setAutoSendBufferSize#250" + cr
523          + "  statechange#<ResourceCheckSum>#<key>=<value>[#<key>=<value>]..." + cr
524          + "    Generates an 'add' command when the 'state' has changed." + cr
525          + "    ResourceCheckSum is an integer that represents the current state of the Resource." + cr
526          + "    This command compares ResourceCheckSum and the Resource field value to the values" + cr

533          + "  quit" + cr
534          + "    Sends any remaining data and exits the sensorshell." + cr
535          + "    Example: quit" + cr;

546      /**
547       * Returns a string with the next line of input from the user. If input
548       * errors, returns the string "quit".
549       *
550       * @return   A string with user input.
551       */
552      private String readLine() {
553        try {
554          String line = this.bufferedReader.readLine();
555          logger.info(line);
556          return (line == null) ? "" : line;
557        }
558        catch (IOException e) {
559          //logger.info(cr);
560          return "quit";
561        }
562      }

691      /**
692       * Implements the "StateChange" algorithm.  The goal of StateChange is to add a new SensorData
693       * instance for sending to the SensorBase only when "something has changed" in the tool (which
694       * is typically an interactive editor or IDE). This method is designed to be used in conjunction
695       * with a timer-based process in the sensor client.  The timer-based process should wake up 
696       * at regular intervals, determine the currently "active" resource (typically a file), and 
697       * compute a "checksum" for that resource representing its state. (This checksum is typically
698       * the size of the resource in bytes or characters.) Having obtained those two values, the 
699       * sensor client can then create a keyValMap as if it were going to do a regular Add command,
700       * and invoke this method with it as well as with the resourceCheckSum.  This method keeps
701       * track of the last Resource and last ResourceCheckSum provided to it, and if either has
702       * changed, it will automatically invoke the Add command, passing the keyValMap to it. 
703       * <p>
704       * Thus, if an editor is running but the user is out at lunch, repeated invocations of the 
705       * StateChange method will not result in any new data being sent to the server.  
706       * @param resourceCheckSum An integer representing the state of the Resource. 
707       * @param keyValMap A map of key-value pairs representing sensor data fields and properties. 
708       * @throws Exception If problems occur during the Add (if the Add actually occurs.)
709       */
710      public synchronized void statechange(long resourceCheckSum, Map<String, String> keyValMap) 
711      throws Exception {
712        this.sensorDataCommand.statechange(resourceCheckSum, keyValMap);
713      }
