024    /**
025     * Provides "middleware" for accumulating and sending notification (sensor)
026     * data to Hackystat. SensorShell has two modes of interaction: command line and
027     * programmatic. 
028     * <p> 
029     * Command line mode is entered by invoking the main() method, and
030     * is intended to be used as a kind of subshell to which commands to add and
031     * send notification data of various types can be sent. The SensorShell can be invoked
032     * without any additional arguments as follows:
033     * 
034     * <pre>java -jar sensorshell.jar</pre>
035     * 
036     * Or you can invoke it with one, two, three, or four additional arguments:
037     * 
038     * <pre>java -jar sensorshell.jar [toolname] [sensor.properties] [no offline] [command file]</pre>
039     * <p>
040     * Programmatic mode involves creating an instance of SensorShell, retrieving the 
041     * appropriate command instance (Ping, Add, etc.) and invoking the appropriate method.
042     *
043     * @author    Philip M. Johnson
044     */

125      /**
126       * Constructs a new SensorShell instance that can be provided with
127       * notification data to be sent eventually to a specific user key and host.
128       * (For testing purposes only, you may want to disable offline data recovery.)
129       *
130       * @param sensorProperties   The sensor properties instance for this run.
131       * @param isInteractive      Whether this SensorShell is being interactively invoked or not.
132       * @param toolName           The invoking tool that is added to the log file name.
133       * @param enableOfflineData  A boolean indicating whether to recover and/or store offline data.
134       * @param commandFile        A file containing shell commands, or null if none provided.
135       */
136      public SensorShell(SensorProperties sensorProperties, boolean isInteractive, String toolName,
137          boolean enableOfflineData, File commandFile) {
138        this.isInteractive = isInteractive;
139        this.toolName = toolName;
140        this.sensorProperties = sensorProperties;
141        this.enableOfflineData = enableOfflineData;
142        this.commandFile = commandFile;
143        this.commandFilePresent = ((commandFile != null));
144        this.client = new SensorBaseClient(sensorProperties.getHackystatHost(), 
145            sensorProperties.getEmail(), sensorProperties.getPassword());
146        // Create the offline directory in the same directory where sensor.properties was found.
147        //OfflineManager.getInstance(sensorProperties.getSensorPropertiesDir());
148        initializeLogger();
149        this.pingCommand = new PingCommand(this, sensorProperties);
150        this.sensorDataCommand = new SensorDataCommand(this, sensorProperties, this.pingCommand, 
151            this.client);
152        this.autoSendCommand = new AutoSendCommand(this, sensorProperties);
153        this.quitCommand = new QuitCommand(this, sensorProperties, this.sensorDataCommand);
154        printBanner(sensorProperties);
155        this.autoSendCommand.initialize();
156        recoverOfflineData();
157    
158        // Now determine whether to read commands from the input stream or from the command file.
159        try {
160          this.bufferedReader = (commandFilePresent) ?
161            new BufferedReader(new FileReader(this.commandFile)) :
162            new BufferedReader(new InputStreamReader(System.in));
163        }
164        catch (IOException e) {
165          this.logger.info(cr);
166        }
167      }

214      /**
215       * Initializes SensorShell logging. All client input is recorded to a log file
216       * in .hackystat/logs/.
217       */
218      private void initializeLogger() {
219        try {
220          // First, create the logs directory, using the same location as sensor.properties if possible.
221          File logDir = (sensorProperties.getSensorPropertiesDir() == null) ?
222              new File(System.getProperty("user.home") + "/.hackystat/logs") :
223              new File(sensorProperties.getSensorPropertiesDir(), "/logs");
224          logDir.mkdirs();
225    
226          // Now set up logging to a file in that directory.
227          this.logger = Logger.getLogger("hackystat.client.cli.SensorShell");
228          this.logger.setUseParentHandlers(false);
229          String fileName = logDir.getAbsolutePath() + "/" + this.toolName + ".%u.log";
230          FileHandler handler = new FileHandler(fileName, 500000, 1, true);
231          handler.setFormatter(this.oneLineFormatter);
232          this.logger.addHandler(handler);
233          // Add a couple of newlines to the log file to distinguish new shell sessions.
234          logger.info(cr + cr);
235        }
236        catch (Exception e) {
237          System.out.println("Error initializing SensorShell logger:\n" + e);
238        }
239      }

243       * Prints out initial information about the SensorShell.
244       *
245       * @param sensorProperties  The sensor properties file.
246       */
247      private void printBanner(SensorProperties sensorProperties) {
248        this.println("Hackystat SensorShell Version: " + getVersion());
249        this.println("SensorShell started at: " + new Date());
250        this.println("Using Sensor Properties in: " + sensorProperties.getAbsolutePath());
251        this.println("Type 'help' for a list of commands.");
252        // Ping the host to determine availability.
253        String host = sensorProperties.getHackystatHost();
254        String email = sensorProperties.getEmail();
255        String password = sensorProperties.getPassword();
256        String availability = SensorBaseClient.isHost(host) ? "available." : "not available";
257        this.println("Host: " + sensorProperties.getHackystatHost() + " is " + availability);
258        String authorized = SensorBaseClient.isRegistered(host, email, password) ? 
259            " authorized " : " not authorized ";
260        this.println("User " + email + " is" + authorized + "to login at this host.");
261      }

264      /**
265       * Process a single input string representing a command.
266       * @param inputString A command as a String.
267       */
268      private void processInputString(String inputString) {
269        // Ignore empty commands. 
270        if ((inputString == null) || ("".equals(inputString))) {
271          return;
272        }
273        // Log the command if we're not running interactively.
274        if (!this.isInteractive) {
275          logger.info("#> " + inputString);
276        }
277        // Process quit command.
278        if ("quit".equals(inputString)) {
279          this.quitCommand.quit();
280          return;
281        }
282    
283        // Process help command.
284        if ("help".equals(inputString)) {
285          this.printHelp();
286          return;
287        }
288    
289        // Process send command.
290        if ("send".equals(inputString)) {
291          this.sensorDataCommand.send();
292          return;
293        }
294        
295        // Process ping command.
296        if ("ping".equals(inputString)) {
297          boolean isPingable = this.pingCommand.isPingable();
298          this.println("Ping of host " + this.sensorProperties.getHackystatHost() + " for user " +
299              this.sensorProperties.getEmail() + 
300              (isPingable ? " succeeded." : " did not succeed"));
301          return;
302        }
303    
304        // Process commands with arguments. 
305        StringTokenizer tokenizer = new StringTokenizer(inputString, this.delimiter);
306        int numTokens = tokenizer.countTokens();
307        // All remaining commands must have arguments. 
308        if (numTokens == 0) {
309          this.println("Error: unknown command or command requires arguments.");
310          return;
311        }
312    
313        // Get the command name and any arguments. 
314        String commandName = tokenizer.nextToken();
315        ArrayList<String> argList = new ArrayList<String>();
316        while (tokenizer.hasMoreElements()) {
317          argList.add(tokenizer.nextToken());
318        }
319        
320        if ("add".equals(commandName)) {
321          // For an Add command, the argument list should be a set of key-value pairs.
322          // So, build the Map of key-value pairs.
323          Map<String, String> keyValMap = new HashMap<String, String>();
324          for (String arg : argList) {
325            int delim = arg.indexOf('=');
326            if (delim == -1) {
327              this.println("Error: can't parse argument string for add command.");
328            }
329            keyValMap.put(arg.substring(0, delim), arg.substring(delim + 1));
330          }
331          try {
332            this.sensorDataCommand.add(keyValMap);
333          }
334          catch (Exception e) {
335            this.println("Error: Can't parse the Timestamp or Runtime arguments.");
336          }
337          return;
338        }
339        
340        if ("autosend".equals(commandName)) {
341          String interval = (argList.isEmpty()) ? "" : argList.get(0);
342          this.autoSendCommand.initialize(interval);
343        }
344      }

347      /**
348       * The command line shell interface.
349       * <ul>
350       *   <li> If invoked with no arguments, then the default sensor.properties
351       *   file is used and the toolName field in the sensor log file name is
352       *   "interactive".
353       *   <li> If invoked with one argument, that argument is used as the toolName
354       *   value.
355       *   <li> If invoked with two arguments, then the first argument is used as
356       *   the toolName value and the second is used as the sensor.properties file
357       *   path.
358       *   <li> If invoked with three arguments, then the first argument is the
359       *   toolname, the second is the sensor.properties file path, and the third
360       *   argument (regardless of value) disables offline data management.
361       *   <li> If invoked with four arguments, then the first argument is the
362       *   toolname, the second is the sensor.properties file path, the third
363       *   argument (regardless of value) disables offline data management, and
364       *   the fourth argument specifies a file of commands (the last of which should be 'quit').
365       * </ul>
366       * Unless four arguments are provided,
367       * the shell then provides a ">>" prompt and supports interactive entry of
368       * sensor data. The following commands are supported:
369       * <ul>
370       *   <li> "help" provides a summary of the available commands.
371       *   <li> "send" sends all of the accumulated data to the server.
372       *   <li> "autosend" sets up a timer-based process that invokes send intermittently.
373       *   <li> "quit" sends all of the accumulated data to the server and exits.
374       *   <li> "ping" checks to see if the host/user/password is valid. 
375       *   <li> "add" adds a single Sensor Data instance to the buffered list to send.
376       * </ul>
377       *
378       * @param args  The command line parameters. See above for details.
379       */
380      public static void main(String args[]) {
381        // Print help line and exit if arg is -help.
382        if ((args.length == 1) && (args[0].equalsIgnoreCase("-help"))) {
383          System.out.println("java -jar sensorshell.jar [toolname] [sensor.properties] [no offline] "
384               + "[command filename]");
385          return;
386        }
387        
388        // Set Parameter 1 (toolname) to supplied or default value.
389        String toolName = (args.length > 0) ? args[0] : "interactive";
390    
391        // Set Parameter 2 (sensor properties file) to supplied or default value. Exit if can't find it.
392        SensorProperties sensorProperties = (args.length >= 2) ?
393            new SensorProperties(new File(args[1])) : new SensorProperties();
394        if (!sensorProperties.isFileAvailable()) {
395          System.out.println("Could not find sensor.properties file. ");
396          System.out.println("Expected in: " + sensorProperties.getAbsolutePath());
397          System.out.println("Exiting...");
398          return;
399        }
400    
401        // Set Parameter 3 (offline). True if we don't supply a value, false if any value supplied.
402        boolean offlineEnabled = ((args.length < 3));
403    
404        // Set Parameter 4 (command file). Null if not supplied. Exit if supplied and bogus.
405        File commandFile = null;
406        if (args.length == 4) {
407          commandFile = new File(args[3]);
408          if (!(commandFile.exists() && commandFile.isFile())) {
409            System.out.println("Could not find the command file. Exiting...");
410            return;
411          }
412        }
413    
414        // Set interactive parameter. From command line, always interactive unless using command file.
415        boolean interactive = ((commandFile == null));
416    
417        // Now create the shell instance, supplying it with all the appropriate arguments.
418        SensorShell shell = 
419          new SensorShell(sensorProperties, interactive, toolName, offlineEnabled, commandFile);
420    
421        // Start processing commands either interactively or from the command file.
422        while (true) {
423          // Get the next command
424          shell.printPrompt();
425          String inputString = shell.readLine();
426          shell.processInputString(inputString);
427          if (inputString.equalsIgnoreCase("quit")) {
428            return;
429          }
430        }
431      }

433      /** Prints the help strings associated with all commands. */
434      private void printHelp() {
435        String helpString = 
436          "SensorShell Command Summary " + cr
437          + "  add#<key>=<value>[#<key>=<value>]..." + cr
438          + "    Adds a new sensor data instance for subsequent sending." + cr 
439          + "    Provide fields and properties as key=value pairs separated by '#'." + cr
440          + "    Owner, Timestamp, and Runtime fields will default to the current user and time." + cr
441          + "    Example: add#Tool=Eclipse#SensorDataType=DevEvent#DevEventType=Compile" + cr 
442          + "  send" + cr
443          + "    Sends any added sensor data to the server. " + cr
444          + "    Server is pinged, and if it does not respond, then data is stored offline." + cr
445          + "    Example: send" + cr
446          + "  ping" + cr
447          + "    Pings the server and checks email/password credentials." + cr
448          + "    Example: ping" + cr
449          + "  autosend#<integer>" + cr
450          + "    Sets the interval in minutes between automatic sending of added sensor data." + cr
451          + "    Provide 0 as the integer to disable autosend." + cr
452          + "    Example: autosend#15" + cr
453          + "  quit" + cr
454          + "    Sends any remaining data and exits the sensorshell." + cr
455          + "    Example: quit" + cr;
456        this.print(helpString);
457      }

466      /**
467       * Returns a string with the next line of input from the user. If input
468       * errors, returns the string "quit".
469       *
470       * @return   A string with user input.
471       */
472      private String readLine() {
473        try {
474          String line = this.bufferedReader.readLine();
475          logger.info(line);
476          return (line == null) ? "" : line;
477        }
478        catch (IOException e) {
479          //logger.info(cr);
480          return "quit";
481        }
482      }

485       * Prints out the line plus newline if in interactive mode, and always logs the line.
486       * Provided to clients to support logging of error messages. 
487       * 
488       * @param line  The line to be printed.
489       */
490      public final void println(String line) {
491        logger.info(line + cr);
492        if (isInteractive) {
493          System.out.print(line + cr);
494        }

518          Package thisPackage = Class.forName("org.hackystat.sensorshell.SensorShell").getPackage();

552      /**
553       * Converts the values in the KeyValMap to a SensorData instance and adds it to the shell.
554       * This is the easiest way to add sensor data.
555       * The Owner will default to the hackystat user in the sensor.properties file.
556       * The Timestamp and Runtime will default to the current time.  
557       * @param keyValMap The map of key-value pairs. 
558       * @throws Exception If the Map cannot be translated into SensorData, typically because a 
559       * value was passed for Timestamp or Runtime that could not be parsed into XMLGregorianCalendar. 
560       */
561      public void add(Map<String, String> keyValMap) throws Exception {
562        this.sensorDataCommand.add(keyValMap);
563      }
