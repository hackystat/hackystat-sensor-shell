020    /**
021     * Provides testing classes and a main method for experimenting with MultiSensorShell performance.
022     * <p>
023     * The main() method provides a simple experimental interface.  To use it, bring up a SensorBase.
024     * Then invoke the main() method, editing the initial local variables as you wish. The system
025     * will run, printing out the elapsed time in milliseconds required for each add() of an 
026     * individual SensorData instance.  At the end of the run, it will print out the total elapsed
027     * time and the average time to send each SensorData instance to the server.  Note that I have
028     * seen a variance of around 0.5 milliseconds in the average time per instance send. So, for
029     * example, when repeatedly running under the same settings, I have obtained values 
030     * varying between 3.1 to about 3.7 milliseconds per instance.  This is with totalData = 5000.
031     * It appears that the variance decreases if you increase totalData to about 50,000. At that 
032     * point, I get relatively consistent throughput of around 2.8 milliseconds per instance. 
033     *
034     * @author Philip Johnson
035     *
036     */
037    public class TestMultiSensorShell {
038      
039      /** The test user. */
040      private static String user = "MultiShell@hackystat.org";
041      private static String host;
042      private static Server server;
043      
044      /**
045       * Starts the server going for these tests, and makes sure our test user is registered. 
046       * @throws Exception If problems occur setting up the server. 
047       */
048      @BeforeClass public static void setupServer() throws Exception {
049        TestMultiSensorShell.server = Server.newInstance();
050        TestMultiSensorShell.host = TestMultiSensorShell.server.getHostName();
051        SensorBaseClient.registerUser(host, user);
052      }
053      
054      /**
055       * Gets rid of the sent sensor data and the user. 
056       * @throws Exception If problems occur setting up the server. 
057       */
058      @AfterClass public static void teardownServer() throws Exception {
059        // Now delete all data sent by this user.
060        SensorBaseClient client = new SensorBaseClient(host, user, user);
061        // First, delete all sensor data sent by this user. 
062        SensorDataIndex index = client.getSensorDataIndex(user);
063        for (SensorDataRef ref : index.getSensorDataRef()) {
064          client.deleteSensorData(user, ref.getTimestamp());
065        }
066        // Now delete the user too.
067        client.deleteUser(user);
068      }
069      
070      /**
071       * Tests that the MultiSensorShell can send data to the Server and this data can be retrieved.
072       * @throws Exception If problems occur. 
073       */
074      @Test public void testMultiSensorShell() throws Exception {
075        // First, start up a SensorShell.
076        SensorProperties properties = new SensorProperties(host, user, user);
077        // Create a MultiSensorShell with default performance properties. 
078        MultiSensorShell shell = new MultiSensorShell(properties, "Test");
079        
080        // Now construct a key-val map representing a SensorData instance. 
081        Map<String, String> keyValMap = new HashMap<String, String>();
082        XMLGregorianCalendar tstamp = Tstamp.makeTimestamp();
083        keyValMap.put("Timestamp", tstamp.toString());
084        String tool = "Eclipse";
085        keyValMap.put("Tool", tool);
086        keyValMap.put("SensorDataType", "DevEvent");
087        keyValMap.put("DevEvent-Type", "Compile");
088        
089        // Now add it to the MultiSensorShell and send it to the server. 
090        shell.add(keyValMap);
091        shell.send();
092        shell.ping();
093        shell.quit();
094        
095        // Now retrieve it from the server using a SensorBaseClient.
096        SensorBaseClient client = new SensorBaseClient(host, user, user);
097        // Now, get the data, dude. 
098        SensorData data = client.getSensorData(user, tstamp);
099        assertEquals("Checking data", tool, data.getTool());
100      }
101      
102      /**
103       * Creates a sample SensorData instance given a user. 
104       * @param user The user.
105       * @return The new SensorData instance. 
106       */
107      private static SensorData makeSensorData(String user) {
108        String sdt = "TestSdt";
109        SensorData data = new SensorData();
110        String tool = "Subversion";
111        data.setTool(tool);
112        data.setOwner(user);
113        data.setSensorDataType(sdt);
114        data.setTimestamp(Tstamp.makeTimestamp());
115        data.setResource("file://foo/bar/baz.txt");
116        data.setRuntime(Tstamp.makeTimestamp());
117        Property property = new Property();
118        property.setKey("SampleField");
119        property.setValue("The test value for Sample Field");
120        Properties properties = new Properties();
121        properties.getProperty().add(property);
122        data.setProperties(properties);
123        return data;
124      }
125      
126      /**
127       * A simple main class that illustrates how to invoke the MultiSensorShell to explore 
128       * performance improvements related to throughput of SensorData instances.  The idea is 
129       * bring up a local SensorBase, edit the local variables of this method, then run this 
130       * main() to see how fast you can transmit the SensorData to the SensorBase. 
131       * @param args Ignored.
132       * @throws Exception If things go wrong. 
133       */
134      public static void main(String[] args) throws Exception {
135        long totalData = 5000;
136        int numShells = 10;
137        double autoSendTimeInterval = 0.05;
138        int autoSendBatchSize = 1000;
139        int batchSize = 200;
140        String host = "http://localhost:9876/sensorbase";
141        String user = "TestUser@hackystat.org";
142        System.out.println("Starting run with: totalData: " + totalData + ", numShells: " + numShells + 
143            ", autoSendTimeInterval: " + autoSendTimeInterval + ", batchSize: " + batchSize);
144        SensorProperties properties = new SensorProperties(host, user, user);
145        MultiSensorShell multiShell = 
146          new MultiSensorShell(properties, "MultiTest", false, autoSendTimeInterval, 
147              autoSendBatchSize, numShells, batchSize);
148        // Make sure the user remembers to get the SensorBase running. :-)
149        if (!multiShell.ping()) {
150          throw new Exception("Before running MultiShell, you probably want to run SensorBase!");
151        }
152        Date startTime = new Date();
153        for (int i = 0; i < totalData; i++) {
154          long time1 = new Date().getTime();
155          SensorData data = makeSensorData(user);
156          multiShell.add(data);
157          long time2 = new Date().getTime();
158          System.out.println("Elapsed millis for add: " + (time2 - time1));
159        }
160        multiShell.quit();
161        long totalTime = (new Date().getTime() - startTime.getTime());
162        System.out.println("Total time: " + totalTime + " milliseconds.");
163        double timePerData = (double)totalTime / (double)totalData;
164        System.out.println("Milliseconds/sensordata instance: " + timePerData);
165    
166      }
