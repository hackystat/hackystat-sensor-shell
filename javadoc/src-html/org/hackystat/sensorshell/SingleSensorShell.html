115      /**
116       * Constructs a new SensorShell instance that can be provided with
117       * notification data to be sent eventually to a specific user key and host.
118       * (For testing purposes only, you may want to disable offline data recovery.)
119       *
120       * @param properties   The sensor properties instance for this run.
121       * @param isInteractive      Whether this SensorShell is being interactively invoked or not.
122       * @param toolName           The invoking tool that is added to the log file name.
123       * @param commandFile        A file containing shell commands, or null if none provided.
124       */
125      public SingleSensorShell(SensorShellProperties properties, boolean isInteractive, String toolName,
126          File commandFile) {
127        this.isInteractive = isInteractive;
128        this.toolName = toolName;
129        this.sensorProperties = properties;
130        this.commandFile = commandFile;
131        this.commandFilePresent = ((commandFile != null));
132        this.client = new SensorBaseClient(properties.getSensorBaseHost(), 
133            properties.getSensorBaseUser(), properties.getSensorBasePassword());
134        this.client.setTimeout(properties.getTimeout() * 1000);
135        initializeLogger();
136        printBanner();
137        this.pingCommand = new PingCommand(this, properties);
138        this.sensorDataCommand = new SensorDataCommand(this, properties, this.pingCommand, 
139            this.client);
140        this.autoSendCommand = new AutoSendCommand(this, properties);
141        this.quitCommand = new QuitCommand(this, properties, sensorDataCommand, autoSendCommand);
142    
143        // Now determine whether to read commands from the input stream or from the command file.
144        try {
145          this.bufferedReader = (commandFilePresent) ?
146            new BufferedReader(new FileReader(this.commandFile)) :
147            new BufferedReader(new InputStreamReader(System.in));
148        }
149        catch (IOException e) {
150          this.logger.info(cr);
151        }
152        this.offlineManager = new OfflineManager(this);
153        try {
154          recoverOfflineData();
155        }
156        catch (SensorShellException e) {
157          this.logger.warning("Error recovering offline data.");
158        }
159      }

170      /**
171       * Looks for offline data and recovers any that is found if offline data
172       * management is enabled and if the server is currently pingable.
173       * @throws SensorShellException If problems occur recovering the data.
174       */
175      private void recoverOfflineData() throws SensorShellException {
176        // Return immediately if server is not available.
177        if (this.pingCommand.isPingable() && this.sensorProperties.isOfflineRecoveryEnabled()) {
178          this.println("Checking for offline data to recover.");
179          this.offlineManager.recover();
180        }
181        else {
182          this.println("Server not available or offline recovery disabled.");
183        }
184      }

187      /**
188       * Initializes SensorShell logging. All client input is recorded to a log file
189       * in [user.home]/.hackystat/sensorshell/logs.  Note that [user.home] is obtained 
190       * from HackystatUserHome.getHome().
191       */
192      private void initializeLogger() {
193        try {
194          // First, create the logs directory.
195          File logDir = new File(HackystatUserHome.getHome() + ".hackystat/sensorshell/logs");
196          logDir.mkdirs();
197    
198          // Now set up logging to a file in that directory.
199          this.logger = Logger.getLogger("org.hackystat.sensorshell");
200          this.logger.setUseParentHandlers(false);
201          String fileName = logDir.getAbsolutePath() + "/" + this.toolName + ".%u.log";
202          FileHandler handler = new FileHandler(fileName, 500000, 1, true);
203          handler.setFormatter(this.oneLineFormatter);
204          this.logger.addHandler(handler);
205          // Add a couple of newlines to the log file to distinguish new shell sessions.
206          logger.info(cr + cr);
207          // Now set the logging level based upon the SensorShell Property.
208          logger.setLevel(this.sensorProperties.getLoggingLevel());
209          logger.getHandlers()[0].setLevel(this.sensorProperties.getLoggingLevel());
210        }
211        catch (Exception e) {
212          System.out.println("Error initializing SensorShell logger:\n" + e);
213        }
214      }

217      /**
218       * Prints out initial information about the SensorShell.
219       */
220      private void printBanner() {
221        this.println("Hackystat SensorShell Version: " + getVersion());
222        this.println("SensorShell started at: " + new Date());
223        this.println(sensorProperties.toString());
224        this.println("Type 'help' for a list of commands.");
225        // Ping the host to determine availability.
226        String host = sensorProperties.getSensorBaseHost();
227        String email = sensorProperties.getSensorBaseUser();
228        String password = sensorProperties.getSensorBasePassword();
229        String availability = SensorBaseClient.isHost(host) ? "available." : "not available";
230        this.println("Host: " + sensorProperties.getSensorBaseHost() + " is " + availability);
231        String authorized = SensorBaseClient.isRegistered(host, email, password) ? 
232            " authorized " : " not authorized ";
233        this.println("User " + email + " is" + authorized + "to login at this host.");
234      }

237      /**
238       * Process a single input string representing a command.
239       * @param inputString A command as a String.
240       * @throws SensorShellException If problems occur sending the data. 
241       */
242      void processInputString(String inputString) throws SensorShellException {
243        // Ignore empty commands. 
244        if ((inputString == null) || ("".equals(inputString))) {
245          return;
246        }
247        // Log the command if we're not running interactively.
248        if (!this.isInteractive) {
249          logger.info("#> " + inputString);
250        }
251        // Process quit command.
252        if ("quit".equals(inputString)) {
253          this.quitCommand.quit();
254          return;
255        }
256    
257        // Process help command.
258        if ("help".equals(inputString)) {
259          this.printHelp();
260          return;
261        }
262    
263        // Process send command.
264        if ("send".equals(inputString)) {
265          this.sensorDataCommand.send();
266          return;
267        }
268        
269        // Process ping command.
270        if ("ping".equals(inputString)) {
271          boolean isPingable = this.pingCommand.isPingable();
272          this.println("Ping of host " + this.sensorProperties.getSensorBaseHost() + " for user " +
273              this.sensorProperties.getSensorBaseUser() + 
274              (isPingable ? " succeeded." : " did not succeed"));
275          return;
276        }
277    
278        // Process commands with arguments. 
279        StringTokenizer tokenizer = new StringTokenizer(inputString, this.delimiter);
280        int numTokens = tokenizer.countTokens();
281        // All remaining commands must have arguments. 
282        if (numTokens == 0) {
283          this.println("Error: unknown command or command requires arguments.");
284          return;
285        }
286    
287        // Get the command name and any arguments. 
288        String commandName = tokenizer.nextToken();
289        ArrayList<String> argList = new ArrayList<String>();
290        while (tokenizer.hasMoreElements()) {
291          argList.add(tokenizer.nextToken());
292        }
293        
294        if ("add".equals(commandName)) {
295          // For an Add command, the argument list should be a set of key-value pairs.
296          // So, build the Map of key-value pairs.
297          Map<String, String> keyValMap = new HashMap<String, String>();
298          for (String arg : argList) {
299            int delim = arg.indexOf('=');
300            if (delim == -1) {
301              this.println("Error: can't parse argument string for add command.");
302            }
303            keyValMap.put(arg.substring(0, delim), arg.substring(delim + 1));
304          }
305          try {
306            this.sensorDataCommand.add(keyValMap);
307          }
308          catch (Exception e) {
309            this.println("Error: Can't parse the Timestamp or Runtime arguments.");
310          }
311          return;
312        }
313        
314        if ("statechange".equals(commandName)) {
315          String resourceCheckSumString = argList.get(0);
316          int resourceCheckSum = 0;
317          try {
318            resourceCheckSum = Integer.parseInt(resourceCheckSumString);
319          }
320          catch (Exception e) {
321            this.println("Error: Can't parse the checksum into an integer.");
322            return;
323          }
324          argList.remove(0);
325          // Now do almost the same as for an add command. 
326          // Build the Map of key-value pairs.
327          Map<String, String> keyValMap = new HashMap<String, String>();
328          for (String arg : argList) {
329            int delim = arg.indexOf('=');
330            if (delim == -1) {
331              this.println("Error: can't parse argument string for statechange command.");
332            }
333            keyValMap.put(arg.substring(0, delim), arg.substring(delim + 1));
334          }
335          try {
336            this.sensorDataCommand.statechange(resourceCheckSum, keyValMap);
337          }
338          catch (Exception e) {
339            this.println("Error: Can't parse the Timestamp or Runtime arguments.");
340          }
341          return;
342        }
343    
344        // Otherwise we don't understand.
345        this.println("Invalid command entered and ignored. Type 'help' for help.");
346      }

350      /** Prints the help strings associated with all commands. */
351      private void printHelp() {
352        String helpString = 
353          "SensorShell Command Summary " + cr
354          + "  add#<key>=<value>[#<key>=<value>]..." + cr
355          + "    Adds a new sensor data instance for subsequent sending." + cr 
356          + "    Provide fields and properties as key=value pairs separated by '#'." + cr
357          + "    Owner, Timestamp, and Runtime fields will default to the current user and time." + cr
358          + "    Example: add#Tool=Eclipse#SensorDataType=DevEvent#DevEventType=Compile" + cr 
359          + "  send" + cr
360          + "    Sends any added sensor data to the server. " + cr
361          + "    Server is pinged, and if it does not respond, then data is stored offline." + cr
362          + "    Example: send" + cr
363          + "  ping" + cr
364          + "    Pings the server and checks email/password credentials." + cr
365          + "    Example: ping" + cr
366          + "  statechange#<ResourceCheckSum>#<key>=<value>[#<key>=<value>]..." + cr
367          + "    Generates an 'add' command when the 'state' has changed." + cr
368          + "    ResourceCheckSum is an integer that represents the current state of the Resource." + cr
369          + "    This command compares ResourceCheckSum and the Resource field value to the values" + cr
370          + "    saved from the last call to statechange.  If either have changed, indicating that" + cr
371          + "    the state has changed, then the key-value pairs are passed to the Add command." + cr
372          + "    This command facilitates the implementation of timer-based sensor processes that" + cr
373          + "    wake up periodically and emit statechange commands, with the knowledge that if " + cr
374          + "    the user has not been active, these statechange commands will not result in" + cr
375          + "    actual sensor data being sent to the server." + cr
376          + "  quit" + cr
377          + "    Sends any remaining data and exits the sensorshell." + cr
378          + "    Example: quit" + cr;
379        this.print(helpString);
380      }

389      /**
390       * Returns a string with the next line of input from the user. If input
391       * errors, returns the string "quit".
392       *
393       * @return   A string with user input.
394       */
395      String readLine() {
396        try {
397          String line = this.bufferedReader.readLine();
398          logger.info(line);
399          return (line == null) ? "" : line;
400        }
401        catch (IOException e) {
402          //logger.info(cr);
403          return "quit";
404        }
405      }

407      /**
408       * Prints out the line plus newline if in interactive mode, and always logs the line.
409       * Provided to clients to support logging of error messages. 
410       * 
411       * @param line  The line to be printed.
412       */
413      public final synchronized void println(String line) {
414        logger.info(line + cr);
415        if (isInteractive) {
416          SimpleDateFormat dateFormat = new SimpleDateFormat("MM/dd HH:mm:ss", Locale.US);
417          System.out.print(dateFormat.format(new Date()) + " " + line + cr);
418        }
419      }

423       * Prints out the line without newline if in interactive mode.

427      private void print(String line) {

429        if (isInteractive) {
430          SimpleDateFormat dateFormat = new SimpleDateFormat("MM/dd HH:mm:ss", Locale.US);
431          System.out.print(dateFormat.format(new Date()) + " " + line);
432        }

435      /**
436       * Return the current version number.
437       *
438       * @return The version number, or "Unknown" if could not be determined.
439       */
440      private String getVersion() {
441        String release;
442        try {
443          Package thisPackage = Class.forName("org.hackystat.sensorshell.SensorShell").getPackage();
444          release = thisPackage.getImplementationVersion();
445        }
446        catch (Exception e) {
447          release = "Unknown";
448        }
449        return release;
450      }
