001    package org.hackystat.sensorshell;
002    
003    
004    /**
005     * An exception that is thrown when SensorProperties can be instantiated.
006     * There are several causes for this exception:
007     * <ol>
008     * <li>Missing sensor properties file</li>
009     * <li>Unparseable file</li>
010     * <li>Read access is denied on the file</>
011     * </ol>
012     * @author Aaron A. Kagawa
013     */
014    public class SensorPropertiesException extends Exception {
015    
016      /** The default serial version UID. */
017      private static final long serialVersionUID = 1L;
018      
019      /**
020       * Thrown when a SensorProperties instance cannot be created.
021       * @param message The message indicating the problem. 
022       */
023      public SensorPropertiesException(String message) {
024        super(message);
025      }
026    }
