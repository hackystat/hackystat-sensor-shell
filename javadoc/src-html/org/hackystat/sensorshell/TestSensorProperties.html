001    package org.hackystat.sensorshell;
002    
003    import static org.junit.Assert.assertEquals;
004    import static org.junit.Assert.assertNotNull;
005    import static org.junit.Assert.fail;
006    
007    import java.io.File;
008    
009    import org.junit.Test;
010    
011    /**
012     * Provides simple unit tests for the SensorProperties.
013     * @author Aaron A. Kagawa
014     */
015    public class TestSensorProperties {
016      
017      /** The test user. */
018      private static String user = "TestShellUser@hackystat.org";
019      private static String host = "TestHost";
020      
021      /**
022       * Tests basic functions of the SensorProperties class.
023       * @throws Exception If problems occur. 
024       */
025      @Test public void testSensorProperties() throws Exception {
026        SensorProperties properties = new SensorProperties(host, user, user);
027        // notice the difference between these two results.
028        assertEquals("Checking hackystat host", TestSensorProperties.host + "/", 
029            properties.getHackystatHost());
030        // how is anyone supposed to know to use HACKYSTAT_SENSORBASE_HOST? 
031        assertEquals("Checking hackystat host", TestSensorProperties.host, 
032            properties.getProperty("HACKYSTAT_SENSORBASE_HOST"));
033        
034        assertEquals("Checking email", TestSensorProperties.user, 
035            properties.getEmail());
036        assertEquals("Checking password", TestSensorProperties.user, 
037            properties.getPassword());
038        assertEquals("Checking auto send time interval", "10", properties.getAutoSendTimeInterval());
039        assertEquals("Checking auto send buffer size", "250", properties.getAutoSendBufferSize());
040        assertEquals("Checking state change interval", 30, properties.getStateChangeInterval());
041        
042        //note that there are a few untestable pieces of code. for example, it is impossible to 
043        //test the default values for Hackystat host, email, and password. It is also impossible to 
044        //test setting of the AutoSendInterval and StateChangeInterval properties.
045      }
046      
047      /**
048       * Tests basic functions of an invalid creation of a SensorProperties instance.
049       * @throws Exception If problems occur. 
050       */
051      @Test public void testInvalidSensorProperties() throws Exception {
052        try {
053          new SensorProperties(null);
054          fail("Should throw an exception");
055        }
056        catch (SensorPropertiesException e) {
057          assertEquals("Checking the exception message", "Invalid sensor properties file", 
058              e.getMessage());
059        }
060        try {
061          new SensorProperties(new File("shoulnd't exist file.wrong extension"));
062          fail("Should throw an exception");
063        }
064        catch (SensorPropertiesException e) {
065          assertNotNull("Checking the exception message", e.getMessage());
066        }
067        
068      }
069    
070      
071    
072    }
