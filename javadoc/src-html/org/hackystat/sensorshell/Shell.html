001    package org.hackystat.sensorshell;
002    
003    import java.util.Map;
004    
005    import org.hackystat.sensorbase.resource.sensordata.jaxb.SensorData;
006    
007    /**
008     * An interface that is implemented by SensorShell and MultiSensorShell. This enables
009     * sensors that can restrict themselves to the use of the following public methods to
010     * easily toggle between the use of SensorShell or MultiSensorShell depending upon 
011     * load requirements.
012     *   
013     * @author Philip Johnson
014     */
015    public interface Shell {
016      
017      /**
018       * Adds the passed SensorData instance to the Shell.
019       * 
020       * @param sensorData The SensorData instance to be queued for transmission.
021       */
022      public void add(SensorData sensorData);
023      
024      /**
025       * Converts the values in the KeyValMap to a SensorData instance and adds it to the
026       * Shell. Owner will default to the hackystat user in the sensor.properties file.
027       * Timestamp and Runtime will default to the current time.
028       * 
029       * @param keyValMap The map of key-value pairs.
030       * @throws Exception If the Map cannot be translated into SensorData, typically because a value
031       * was passed for Timestamp or Runtime that could not be parsed into XMLGregorianCalendar.
032       */
033      public void add(Map<String, String> keyValMap) throws Exception;
034      
035      /**
036       * Returns true if the host can be pinged and the email/password combination is valid.
037       * 
038       * @return True if the host can be pinged and the user credentials are valid.
039       */
040      public boolean ping();
041      
042      /**
043       * Immediately invokes send() on this Shell. Note that you will rarely want
044       * to invoke this method. Instead, during normal operation you will rely on the
045       * autoSendTimeInterval to invoke send() in a separate thread, and then invoke quit() to invoke
046       * send() at the conclusion of the run.
047       * @return The total number of instances sent by Shell. 
048       */
049      public int send();
050      
051      /**
052       * Invokes quit() on this Shell, which invokes a final send() and closes any
053       * logging files.
054       */
055      public void quit();
056      
057      /**
058       * Sets the logging level to be used for this Shell.
059       * If the passed string cannot be parsed into a Level, then INFO is set by default.
060       * @param level The new Level.
061       */
062      public void setLoggingLevel(String level);
063    
064    }
