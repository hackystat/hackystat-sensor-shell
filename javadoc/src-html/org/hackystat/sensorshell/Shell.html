001    package org.hackystat.sensorshell;
002    
003    import java.util.Map;
004    
005    import org.hackystat.sensorbase.resource.sensordata.jaxb.SensorData;
006    
007    /**
008     * An interface that is implemented by SensorShell and MultiSensorShell. This enables
009     * sensors that can restrict themselves to the use of the following public methods to
010     * easily toggle between the use of SensorShell or MultiSensorShell depending upon 
011     * load requirements.
012     * 
013     * As a simple example, a sensor might contain code similar to the following:
014     * <pre>
015     * boolean useMulti = isMultiSensorShellRequested();
016     * Shell shell = (useMulti) ? new MultiSensorShell(...) : new SensorShell(...);
017     *  :
018     * shell.add(...)
019     *  :
020     * shell.quit()
021     * </pre>
022     * Thus, the decision to use a SensorShell vs. a MultiSensorShell can be made at run-time.
023     *   
024     * @author Philip Johnson
025     */
026    public interface Shell {
027      
028      /**
029       * Adds the passed SensorData instance to the Shell.
030       * 
031       * @param sensorData The SensorData instance to be queued for transmission.
032       */
033      public void add(SensorData sensorData);
034      
035      /**
036       * Converts the values in the KeyValMap to a SensorData instance and adds it to the
037       * Shell. Owner will default to the hackystat user in the sensor.properties file.
038       * Timestamp and Runtime will default to the current time.
039       * 
040       * @param keyValMap The map of key-value pairs.
041       * @throws Exception If the Map cannot be translated into SensorData, typically because a value
042       * was passed for Timestamp or Runtime that could not be parsed into XMLGregorianCalendar.
043       */
044      public void add(Map<String, String> keyValMap) throws Exception;
045      
046      /**
047       * Returns true if the host can be pinged and the email/password combination is valid.
048       * 
049       * @return True if the host can be pinged and the user credentials are valid.
050       */
051      public boolean ping();
052      
053      /**
054       * Immediately invokes send() on this Shell. Note that you will rarely want
055       * to invoke this method. Instead, during normal operation you will rely on the
056       * autoSendTimeInterval to invoke send() in a separate thread, and then invoke quit() to invoke
057       * send() at the conclusion of the run.
058       * @return The total number of instances sent by Shell. 
059       */
060      public int send();
061      
062      /**
063       * Invokes quit() on this Shell, which invokes a final send() and closes any
064       * logging files.
065       */
066      public void quit();
067      
068      /**
069       * Sets the logging level to be used for this Shell.
070       * If the passed string cannot be parsed into a Level, then INFO is set by default.
071       * @param level The new Level.
072       */
073      public void setLoggingLevel(String level);
074    
075    }
