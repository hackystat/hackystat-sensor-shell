001    package org.hackystat.sensorshell.command;
002    
003    import org.hackystat.sensorbase.client.SensorBaseClient;
004    import org.hackystat.sensorshell.SensorProperties;
005    import org.hackystat.sensorshell.SensorShell;
006    
007    /**
008     * Implements the Ping command, which ensures that the SensorBase is reachable. 
009     * @author Philip Johnson
010     */
011    public class PingCommand extends Command {
012      
013      /**
014       * Does a ping on the hackystat server and returns true if the server was
015       * accessible. A ping-able server indicates the data will be sent to it,
016       * while a non-pingable server indicates that data will be stored offline.
017       *
018       * @return   True if the server could be pinged.
019       */
020      public boolean isPingable() {
021        return this.isPingable(5000);
022      }
023    
024      /**
025       * Does a ping on the hackystat server and returns true if the server was
026       * accessible. A ping-able server indicates the data will be sent to it,
027       * while a non-pingable server indicates that data will be stored offline.
028       * If the server is not reachable, or does not respond with given time frame, false will
029       * be returned.
030       *
031       * @param milliSecondsToWait Maximum seconds to wait for server response. A 0 value or negative
032       * value is equivalent to set time out to infinity.
033       * @return   True if the server could be pinged.
034       */
035      public boolean isPingable(int milliSecondsToWait) {
036        boolean result = false;
037        int waitTime = milliSecondsToWait <= 0 ? 0 : milliSecondsToWait;
038        PingWorkerThread workThread = new PingWorkerThread(this.host, this.email, this.password);
039        workThread.start();
040        try {
041          workThread.join(waitTime);  //block this thread until work thread dies or times out.
042        }
043        catch (InterruptedException ex) {
044          //do nothing
045        }
046        //if work thread is still alive, then it's time out, result = false by default.
047        if (!workThread.isAlive()) {
048          result = workThread.serverPingable;
049        }
050        return result;
051      }
052    
053      /**
054       * Worker thread to ping the server to determine whether it's reachable or not. The original
055       * ping command is implemented as a synchronous command, a separate thread is need to
056       * implement time out feature.
057       *
058       * @author Qin ZHANG
059       */
060      private static class PingWorkerThread extends Thread {
061    
062        /**
063         * This instance will only be be accessed in the parent thread after the termination of this
064         * thread, there is no need to synchronize access.
065         */
066        private boolean serverPingable = false;
067        private String host;
068        private String email;
069        private String password;
070    
071        /**
072         * Constructs this worker thread.
073         * @param host The sensorbase host.
074         * @param email The client email.
075         * @param password The client password.
076         */
077        public PingWorkerThread(String host, String email, String password) {
078          setDaemon(true); //want VM to termine even if this thread is alive.
079          this.host = host;
080          this.email = email;
081          this.password = password;
082        }
083    
084        /** Pings the server synchronously. */
085        @Override
086        public void run() {
087          this.serverPingable = SensorBaseClient.isRegistered(this.host, this.email, this.password);
088        }
089      }
090    
091      /**
092       * Creates the PingCommand. 
093       * @param shell The sensorshell. 
094       * @param properties The sensorproperties. 
095       */
096      public PingCommand(SensorShell shell, SensorProperties properties) {
097        super(shell, properties);
098      }
099    }
