001    package org.hackystat.sensorshell.command;
002    
003    import java.util.logging.Handler;
004    
005    import org.hackystat.sensorshell.SensorProperties;
006    import org.hackystat.sensorshell.SensorShell;
007    
008    /**
009     * Implements the Quit command, which sends any buffered data and closes the loggers. 
010     * @author Philip Johnson
011     */
012    public class QuitCommand extends Command {
013      
014      /** Holds the sensor data command for sending data. */
015      private SensorDataCommand sensorDataCommand;
016      
017      /**
018       * Creates the QuitCommand. 
019       * @param shell The sensorshell. 
020       * @param properties The sensorproperties.
021       * @param sensorDataCommand The SensorDataCommand. 
022       */
023      public QuitCommand(SensorShell shell, SensorProperties properties, 
024          SensorDataCommand sensorDataCommand) {
025        super(shell, properties);
026        this.sensorDataCommand = sensorDataCommand;
027      }
028      
029      /** Quits the shell. Sends all data and closes the loggers. */
030      public void quit() {
031        // Log this command if not running interactively.
032        if (!this.shell.isInteractive()) {
033          this.shell.getLogger().info("#> quit" + cr);
034        }
035        this.sensorDataCommand.send();
036        // Now close all loggers. 
037        if (this.shell.getLogger() != null) {
038          // Close all the open handler.
039          Handler[] handlers = this.shell.getLogger().getHandlers();
040          for (int i = 0; i < handlers.length; i++) {
041            handlers[i].close();
042          }
043        }
044        this.shell.println("Quitting.");
045      }
046    }
