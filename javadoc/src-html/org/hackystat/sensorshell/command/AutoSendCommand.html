001    package org.hackystat.sensorshell.command;
002    
003    import java.util.Timer;
004    import java.util.TimerTask;
005    
006    import org.hackystat.sensorshell.SensorProperties;
007    import org.hackystat.sensorshell.SensorShell;
008    
009    /**
010     * Implements the AutoSend facility, which automatically sends all SensorData to the Sensorbase
011     * at regular intervals as specified in the sensor.properties file.  
012     * @author Philip Johnson
013     */
014    public class AutoSendCommand extends Command {
015      
016      /**
017       * Creates the AutoSendCommand. 
018       * @param shell The sensorshell. 
019       * @param properties The sensorproperties. 
020       */
021      public AutoSendCommand(SensorShell shell, SensorProperties properties) {
022        super(shell, properties);
023      }
024      
025      /** The timer that enables periodic sending. */
026      private Timer timer = null;
027      
028      /**
029       * Sets the AutoSend interval from the sensor.properties file. 
030       */
031      public void initialize() {
032        initialize(this.properties.getAutoSendInterval());
033      }
034    
035      /**
036       * Process an AutoSend command, given a list whose first argument is the auto
037       *
038       * @param interval  An integer that specifies the interval in minutes for sending data.
039       */
040      public void initialize(String interval) {
041        // Begin by getting the number of minutes
042        int minutes = 0;
043        try {
044          minutes = Integer.parseInt(interval);
045          if (minutes < 0) {
046            throw new IllegalArgumentException("Minutes must be a non-negative integer.");
047          }
048        }
049        catch (Exception e) {
050          this.shell.println("AutoSend error (invalid argument: " + interval + ")" + cr + e);
051        }
052        initialize(minutes);
053      }
054      
055      /**
056       * Initializes the autosend to the number of specified minutes. If 0, then autosend is disabled. 
057       * @param minutes The minutes. 
058       */
059      public void initialize(int minutes) {
060        // Cancel the current timer if any.
061        if (this.timer != null) {
062          this.timer.cancel();
063        }
064        // Don't set up a new timer if minutes value is 0.
065        if (minutes == 0) {
066          this.shell.println("AutoSend disabled");
067        }
068        else {
069          // Otherwise set up a timer with the newly specified value.
070          this.timer = new Timer(true);
071          int milliseconds = minutes * 60 * 1000;
072          this.timer.schedule(new AutoSendCommandTask(shell), milliseconds, milliseconds);
073          this.shell.println("AutoSend set to " + minutes + " minutes");
074        }
075      }
076      
077      /**
078       * Inner class providing a timer-based command to invoke the send() method of the SensorShell.
079       * @author Philip M. Johnson
080       */
081      private static class AutoSendCommandTask extends TimerTask {
082        
083        /** The sensor shell. */
084        private SensorShell shell;
085    
086        /**
087         * Creates the TimerTask.
088         * @param shell The sensorshell.
089         */
090        public AutoSendCommandTask(SensorShell shell) {
091          this.shell = shell;
092        }
093    
094        /** Invoked periodically by the timer in AutoSendCommand. */
095        @Override
096        public void run() {
097          this.shell.println("Autosending...");
098          this.shell.send();
099        }
100      }
101    }
