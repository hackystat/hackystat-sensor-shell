001    package org.hackystat.sensorshell.command;
002    
003    import java.util.Timer;
004    import java.util.TimerTask;
005    
006    import org.hackystat.sensorshell.SensorShellException;
007    import org.hackystat.sensorshell.SensorShellProperties;
008    import org.hackystat.sensorshell.SingleSensorShell;
009    
010    /**
011     * Implements the AutoSend facility, which automatically sends all SensorData to the Sensorbase
012     * at regular intervals as specified in the sensorshell.autosend.timeinterval property.
013     * @author Philip Johnson
014     */
015    public class AutoSendCommand extends Command {
016      
017      /** The timer that enables periodic sending. */
018      private Timer timer = null;
019      
020      /**
021       * Creates the AutoSendCommand and starts a timer-based process running that wakes up and 
022       * invokes send based upon the value of the SensorShellProperties autosend timeinterval.
023       * @param shell The sensorshell. 
024       * @param properties The sensorproperties. 
025       */
026      public AutoSendCommand(SingleSensorShell shell, SensorShellProperties properties) {
027        super(shell, properties);
028        double minutes = properties.getAutoSendTimeInterval();
029        // Don't set up a timer if minutes value is close to 0.
030        if (minutes < 0.009) {
031          this.shell.println("AutoSend disabled.");
032        }
033        else {
034          // Otherwise set up a timer with the newly specified value.
035          this.timer = new Timer(true);
036          int milliseconds = (int) (minutes * 60 * 1000);
037          this.timer.schedule(new AutoSendCommandTask(shell), milliseconds, milliseconds);
038          this.shell.println("AutoSend time interval set to " + (int) (minutes * 60) + " seconds");
039        }
040      }
041      
042      /**
043       * Cancels the timer if there is one. 
044       */
045      public void quit() {
046        if (this.timer != null) {
047          this.timer.cancel();
048        }
049      }
050      
051      /**
052       * Inner class providing a timer-based command to invoke the send() method of the SensorShell.
053       * @author Philip M. Johnson
054       */
055      private static class AutoSendCommandTask extends TimerTask {
056        
057        /** The sensor shell. */
058        private SingleSensorShell shell;
059    
060        /**
061         * Creates the TimerTask.
062         * @param shell The sensorshell.
063         */
064        public AutoSendCommandTask(SingleSensorShell shell) {
065          this.shell = shell;
066        }
067    
068        /** Invoked periodically by the timer in AutoSendCommand. */
069        @Override
070        public void run() {
071          this.shell.println("Autosending...");
072          try {
073            this.shell.send();
074          }
075          catch (SensorShellException e) {
076            this.shell.println("Error sending data.");
077          }
078        }
079      }
080    }
