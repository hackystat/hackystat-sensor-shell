016    /**
017     * Implements the SensorData commands, of which there is "add", "send", and "statechange".
018     * @author Philip Johnson
019     */
020    public class SensorDataCommand extends Command { 
021        
022      private static final String RESOURCE = "Resource";
023      /** The list of unsent SensorData instances. */
024      private SensorDatas sensorDatas = new SensorDatas();
025      /** The Ping Command. */
026      private PingCommand pingCommand;
027      /** The sensorbase client. */
028      private SensorBaseClient client;
029      /** Holds the Resource value from the last StateChange event. */
030      private String lastStateChangeResource = "";
031      /** Holds the bufferSize value from the last StateChange event. */
032      private long lastStateChangeResourceCheckSum = 0;
033      /** The maximum number of sensor data instances to buffer locally before sending. */
034      private int maxBufferSize = 250;
035      
036      /**
037       * Creates the SensorDataCommand. 
038       * @param shell The sensorshell. 
039       * @param properties The sensorproperties.
040       * @param pingCommand The Ping Command. 
041       * @param client The SensorBase client.
042       */
043      public SensorDataCommand(SensorShell shell, SensorProperties properties, 
044          PingCommand pingCommand, SensorBaseClient client) {
045        super(shell, properties);
046        this.pingCommand = pingCommand;
047        this.client = client;
048      }
049      
050      /**
051       * Sends accumulated data, including offline and current data from the AddCommand.
052       * If server not pingable, then the offline data is saved for a later attempt.
053       * @return The number of sensor data instances that were sent. 
054       */
055      public int send() {
056        int numDataSent = 0;
057        // Log this command if not running interactively.
058        if (!this.shell.isInteractive()) {
059          this.shell.getLogger().info("#> send" + cr);
060        }
061        if (this.pingCommand.isPingable()) {
062          //OfflineManager.getInstance().clear();
063          try {
064            if (sensorDatas.getSensorData().isEmpty()) {
065              this.shell.println("No sensor data to send.");
066              return 0;
067            }
068            this.client.putSensorDataBatch(sensorDatas);
069            this.shell.println(sensorDatas.getSensorData().size() + " SensorData instances sent to " +
070                this.properties.getHackystatHost());
071            numDataSent = sensorDatas.getSensorData().size();
072            this.sensorDatas.getSensorData().clear();
073          }
074          catch (SensorBaseClientException e) {
075            this.shell.println("Error sending data: " + e);
076            numDataSent = 0;
077          }
078        }
079        else {
080          if (this.shell.enableOfflineData()) {
081            this.shell.println("Server not available. Storing commands offline.");
082            numDataSent = 0;
083            this.shell.getOfflineManager().store(this.sensorDatas);
084          }
085          else {
086            this.shell.println("Server not available and offline storage disabled. Data lost.");
087            numDataSent = 0;
088          }
089        }
090        return numDataSent;
091      }
092      
093      /**
094       * Sets the maximum buffer size for sensor data before automatically invoking send. 
095       * @param maxBufferSize The maximum buffer size. 
096       */
097      public void setBufferSize(int maxBufferSize) {
098        this.maxBufferSize = maxBufferSize;
099      }
100      
101      /**
102       * Returns the data instance in a formatted string.
103       * @param data The sensor data instance. 
104       * @return A string displaying the instance.
105       */
106      private String formatSensorData(SensorData data) {
107        StringBuffer buffer = new StringBuffer(75);
108        buffer.append('<');
109        buffer.append(data.getTimestamp());
110        buffer.append(' ');
111        buffer.append(data.getSensorDataType());
112        buffer.append(' ');
113        buffer.append(data.getOwner());
114        buffer.append(' ');
115        buffer.append(data.getTool());
116        buffer.append(' ');
117        buffer.append(data.getResource());
118        buffer.append(' ');
119        buffer.append(data.getRuntime());
120        buffer.append(' ');
121        for (Property property : data.getProperties().getProperty()) {
122          buffer.append(property.getKey());
123          buffer.append('=');
124          buffer.append(property.getValue());
125          buffer.append(' ');
126        }
127        buffer.append('>');
128        return buffer.toString();
129      }
130     
131      
132      /**
133       * Given a Map containing key-value pairs corresponding to SensorData fields and properties,
134       * constructs a SensorData instance and stores it for subsequent sending to the SensorBase.
135       * @param keyValMap The map of key-value pairs. 
136       * @throws Exception If problems occur creating the SensorData instance. 
137       */
138      public void add(Map<String, String> keyValMap) throws Exception {
139        // Begin by creating the sensor data instance. 
140        SensorData data = new SensorData();
141        XMLGregorianCalendar tstamp = Tstamp.makeTimestamp();
142        data.setOwner(getMap(keyValMap, "Owner", this.properties.getEmail()));
143        data.setResource(getMap(keyValMap, RESOURCE, ""));
144        data.setRuntime(Tstamp.makeTimestamp(getMap(keyValMap, "Runtime", tstamp.toString())));
145        data.setSensorDataType(getMap(keyValMap, "SensorDataType", ""));
146        data.setTimestamp(Tstamp.makeTimestamp(getMap(keyValMap, "Timestamp", tstamp.toString())));
147        data.setTool(getMap(keyValMap, "Tool", "unknown"));
148        data.setProperties(new Properties());
149        // Add all non-standard key-val pairs to the property list. 
150        for (Map.Entry<String, String> entry : keyValMap.entrySet()) {
151          Property property = new Property();
152          String key = entry.getKey();
153          if (isProperty(key)) {
154            property.setKey(key);
155            property.setValue(entry.getValue());
156            data.getProperties().getProperty().add(property);
157          }
158        }
159        add(data);
160      }
161      
162      /**
163       * Adds the SensorData instance, invoking send if the max buffer size has been exceeded.
164       * @param data The SensorData instance to be added.
165       */
166      public void add(SensorData data) {
167        sensorDatas.getSensorData().add(data);
168        this.shell.println("Adding: " + formatSensorData(data));
169        
170        // If that makes the buffer size too big, then send this data. 
171        if (sensorDatas.getSensorData().size() > this.maxBufferSize) {
172          this.shell.println("Invoking send() due to buffer size > " + this.maxBufferSize);
173          this.send();
174        }
175      }
176      
177      /**
178       * Provides an easy way for sensors to implement "StateChange" behavior.  From the sensor side,
179       * StateChange can be implemented by creating a timer process that wakes up at regular 
180       * intervals and gets the file (resource) that the current active buffer is attached to, plus
181       * a "checksum" (typically the size of the resource in characters or bytes or whatever).  
182       * Then, the timer process simply creates the
183       * appropriate keyValMap as if for an 'add' event, and provides it to this method along with 
184       * the buffer size.  This method will check the passed buffer size and the Resource field 
185       * against the values for these two fields passed in the last call of this method, and if 
186       * either one has changed, then the "add" method is called with the keyValMap.  
187       * @param resourceCheckSum Indicates the state of the resource, typically via its size. 
188       * @param keyValMap The map of key-value pairs. 
189       * @throws Exception If problems occur while invoking the 'add' command. 
190       */
191      public void statechange(long resourceCheckSum, Map<String, String> keyValMap) throws Exception {
192        // Get the Resource attribute, default it to "" if not found.
193        String resource = (keyValMap.get(RESOURCE) == null) ? "" : keyValMap.get(RESOURCE);
194        // Do an add if the resource or buffer size has changed.
195        if (!this.lastStateChangeResource.equals(resource)) { //NOPMD
196          this.shell.println("Invoking add: Resource has changed to: " + resource);
197          this.add(keyValMap);
198        }
199        else if (this.lastStateChangeResourceCheckSum != resourceCheckSum) { //NOPMD
200          this.shell.println("Invoking add: CheckSum has changed to: " + resourceCheckSum);
201          this.add(keyValMap);
202        }
203        else {
204          this.shell.println("No change in resource: " + resource + ", checksum: " + resourceCheckSum);
205        }
206        // Always update the 'last' values.
207        this.lastStateChangeResourceCheckSum = resourceCheckSum;
208        this.lastStateChangeResource = resource;
209      }

211      /**
212       * Returns the value associated with key in keyValMap, or the default if the key is not present.
213       * @param keyValMap The map
214       * @param key The key
215       * @param defaultValue The value to return if the key has no mapping.
216       * @return The value to be used.
217       */
218      private String getMap(Map<String, String> keyValMap, String key, String defaultValue) {
219        return (keyValMap.get(key) == null) ? defaultValue : keyValMap.get(key);
220      }
221      
222      /**
223       * Returns true if the passed key is not one of the standard Sensor Data fields including
224       * "Timestamp", "Runtime", "Tool", "Resource", "Owner", or "SensorDataType".
225       * @param key The key.
226       * @return True if the passed key indicates a property, not a standard sensor data field.
227       */
228      private boolean isProperty(String key) {
229        return 
230            (!"Timestamp".equals(key)) &&
231            (!"Runtime".equals(key)) &&
232            (!"Tool".equals(key)) &&
233            (!RESOURCE.equals(key)) &&
234            (!"Owner".equals(key)) &&
235            (!"SensorDataType".equals(key));
236      }
